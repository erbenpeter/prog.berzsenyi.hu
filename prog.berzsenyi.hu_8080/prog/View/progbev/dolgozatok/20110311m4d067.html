4. feladat megoldása

program f4;
{rácspontok az R sugarú körben}
var R,x,y,db: longint;
begin
    db := 0;
    readln(R);
    for x := -R to R do
        for y := -R to R do
            if x*x+y*y < R*R
                then db := db +1;
    writeln(db);
end.
