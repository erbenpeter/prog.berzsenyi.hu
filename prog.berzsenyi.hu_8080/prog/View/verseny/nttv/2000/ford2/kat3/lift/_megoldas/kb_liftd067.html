using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

namespace Lift
{
    class Program
    {
        private int n, k;
        private int ppl = 0;
        private List<int>[] emelet;
        private List<int> lift;
        
        public Program()
        {
            StreamReader ins = new StreamReader("lift39.be");
            StreamWriter outs = new StreamWriter("lift39.ki");
            string[] line = ins.ReadLine().Split(' ');
            n = int.Parse( line[0] );
            k = int.Parse( line[1] );
            emelet = new List<int>[n];
            lift = new List<int>(k);
            for (int i = 0; i < n; i++)
            {
                line = ins.ReadLine().Split(' ');
                emelet[i] = new List<int>(line.Length-1); // 0-át a végén nem számítjuk
                for (int j = 0; j < (line.Length - 1); j++)
                {
                    emelet[i].Add( int.Parse(line[j])-1 );
                    ppl++;
                }
                emelet[i].Sort();
            }

            outs.WriteLine( solve() );
            outs.Close();
        }

        private int solve()
        {
            int loc = 0;
            int moves = 0;
            int dir = 1;
            while (ppl != 0 || loc != 0)
            {
                if (loc == 0 && ppl != 0)
                {
                    // legalul vagyunk, de még van kit szállítani
                    moves++; // megyünk még 1 kört.
                }

                // kitesszük, aki megérkezett.
                for (int i = 0; i < lift.Count; i++)
                {
                    if (lift[i] == loc)
                    {
                        lift.RemoveAt(i);
                        i--; // kivettünk egyet.
                    }
                }

                if (loc + 1 == n)
                {
                    // legtetején vagyunk.
                    dir = -1; // fordul a kocka.
                }
                else if (loc == 0)
                {
                    dir = 1; // felfelé megyünk!
                }

                // bevesszük, aki a jó irányba tart, illetve, befér. Az jön először aki kis távolságra megy.
                if (lift.Count != k && dir == 1)
                {
                    // felfelé haladunk -> lehető legközelebbi, tehát a legkisebb szintre tartó, de a jelenleginél nagyobb poziciójú kell.
                    for (int i = 0; i < emelet[loc].Count; i++)
                    {
                        // végignézzük az adott szinten lévő embereket, növekvő sorrendben.

                        if (emelet[loc][i] == loc)
                        {
                            // ha véletlen lenne olyan, hogy az emberke már az adott szinten van :)
                            emelet[loc].RemoveAt(i);
                            ppl--;
                            i--;
                            continue;
                        }

                        if (emelet[loc][i] > loc) // nagyobb helyre tart mint mi :)
                        {
                            lift.Add(emelet[loc][i]); //liftbe be
                            emelet[loc].RemoveAt(i); //emeltről ki
                            ppl--;
                            i--;
                            if (lift.Count == k) break; // ha tele vagyunk, akkor megyünk tovább.
                        }
                    }
                }
                if (lift.Count != k && dir == -1)
                {
                    // lefelé haladunk
                    for (int i = emelet[loc].Count-1; i >= 0; i--)
                    {
                        // végignézzük az adott szinten lévő embereket, csökkenő sorrendben.

                        if (emelet[loc][i] == loc)
                        {
                            // ha véletlen lenne olyan, hogy az emberke már az adott szinten van :)
                            emelet[loc].RemoveAt(i);
                            ppl--;
                            continue;
                        }

                        if (emelet[loc][i] < loc) // kisebb helyre tart mint mi.
                        {
                            lift.Add(emelet[loc][i]); //liftbe be
                            emelet[loc].RemoveAt(i); //emeltről ki
                            ppl--;
                            if (lift.Count == k) break; // ha tele vagyunk, akkor megyünk tovább.
                        }
                    }
                }

                loc += dir; // mozgunk.
            }

            return moves;
        }

        static void Main(string[] args)
        {
            new Program();
        }
    }
}
