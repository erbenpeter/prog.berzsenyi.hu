Imports System

Module Szamolos

   Const N As Byte = 5
   Const MaxHossz As Byte = 5

   Private cache(N) As Integer

   Private Function Bont(ByVal halmaz As Integer) As Integer
      If cache(halmaz) > 0 Then
         Return cache(halmaz)
      Else
         If halmaz <= 1 Then
            cache(halmaz) = 1
         Else
            Dim sum As Integer = 0

            For i As Byte = 1 To MaxHossz
               If halmaz - i < 0 Then Exit For

               sum += Bont(halmaz - i)
            Next

            cache(halmaz) = sum
         End If

         Return cache(halmaz)
      End If
   End Function

   Public Sub Main()
      Console.WriteLine(Bont(N))
      Console.ReadLine()
   End Sub

End Module
