program kevin;

var i,j:longint; //ciklusvaltozo
a:array[1..137,1..137] of integer; //szomszedsagi maxtix
c:integer; //csuuuucsok szama
e:longint; //eeelek szama
t:text; //a fileunk
x,y:integer; //eelek szaama
apa:array[1..137] of integer; //kinek ki a papaja
tav:array[1..137] of integer; //kinek mi a tavolsaga
start,cel:integer; //honnan indulunk, hova menjunk
sor:array[1..133337] of integer; s1,s2:longint; //sor, kezdo es vegpontja

begin
assign(t,'graf.be'); reset(t); readln(t,c,e);
for i:=1 to c do begin for j:=1 to c do begin a[i,j]:=0; end; end; //nullaz
for i:=1 to c do begin apa[i]:=0; tav[i]:=0; end;
for i:=1 to e do begin readln(t,x,y); a[x,y]:=1; a[y,x]:=1; end; //szom matr
close(t);

write('Honnan induljunk?: ');
readln(start);                                                  //starthely
write('Hova menjünk?: ');
readln(cel);
s1:=1; s2:=1;
sor[s2]:=start; apa[sor[s2]]:=sor[s2]; tav[sor[s2]]:=0;

while s1<=s2 do begin                                           //bejaras
for i:=1 to c do if
(apa[i] = 0) and (a[i,sor[s1]]>0) then begin  {ha szomszed, es apatlan}
apa[i]:=sor[s1]; tav[i]:=tav[sor[s1]]+1; inc(s2); sor[s2]:=i; {o az apam!}
end;
inc(s1); {kovetkezo sor elemet nezem}
end;

writeln(tav[cel],' távol vannak.');


end.
