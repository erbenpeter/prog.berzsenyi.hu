using System;
using System.Collections;
using System.IO;

namespace PruferKod
{
    
    class El {
        public int j;
        public int k;
        
        public El( int j, int k ) {
            this.j = j;
            this.k = k;
        }
    }
    
    class Program
    {
        
        public int[] prufer;
        public static ArrayList prufers;
        public El[] ellista;
        public int[,] szommatrix;
        public int[,] d;
        
        public void doElLista() {
            //Legeneráljuk az Ellistat
            ellista = new El[prufer.Length+1];
            ArrayList szamok;
            ArrayList beirt = new ArrayList();
            
            for( int i=0; i<prufer.Length; i++ ) {
                // 1. megvizsgáljuk, hogy melyik az a legkisebb szám, ami nincs benne a prüferben.
                // 2. ezt a számot, hozzákötjük a prufer[i]-hez.
                szamok = new ArrayList();
                int a;
                for( a=i; a<prufer.Length; a++ ) {
                    if( !szamok.Contains( prufer[a] ) ) {
                        szamok.Add( prufer[a] );
                    }
                }
                
                for( a=1; a<=(prufer.Length+2); a++ ) {
                    if( !szamok.Contains( a ) && !beirt.Contains( a ) ) {
                        break;
                    }
                }
                
                ellista[i] = new El( a, prufer[i] );
                beirt.Add( a );
                
                // --> a <-- ezt a számot kötjük összel prufer[i]-vel.
                
            }
            
            int b;
            for( b=1; b<=(prufer.Length+2); b++ ) {
                if( !beirt.Contains( b ) ) {
                    break;
                }
            }
            
            ellista[prufer.Length] = new El( b, prufer.Length+2 );
                        
        }
        
        public void doSzomszedsagiMatrix() {
            szommatrix = new int[ellista.Length+1, ellista.Length+1];
            for( int x=0; x<ellista.Length+1; x++ ) {
                for( int y=0; y<ellista.Length+1; y++ ) {
                    szommatrix[x, y] = 0;
                }
            }
            for( int i=0; i<ellista.Length; i++ ) {
                szommatrix[ ellista[i].j-1, ellista[i].k-1 ] = 1;
                szommatrix[ ellista[i].k-1, ellista[i].j-1 ] = 1;
            }
        }
        
        private void doD( int csucs, int b, int a ) {
            
            for( int i=0; i<ellista.Length+1; i++ ) {
                if( szommatrix[ b, i ] == 1 ) {
                    if( d[ csucs, i ] == 0 && (csucs != i) ) {
                        d[ csucs, i ] = a;
                        doD( csucs, i, a+1 );
                    }
                }
            }
            
        }
        
        public void calcAtmero() {
            // végig megyünk a csúcsokon (tehát a szommatrixon.)
            // bejelöljük a d(csucs, j)-khez, hogy mennyi ido.
                        
            d = new int[ellista.Length+1, ellista.Length+1];
            
            for( int x=0; x<ellista.Length+1; x++ ) {
                for( int y=0; y<ellista.Length+1; y++ ) {
                    d[ x, y ] = 0;
                }   
            }
            
            int max = 0;            
            for( int csucs=0; csucs<(ellista.Length+1); csucs++ ) {
                // csucs megvan.
                doD( csucs, csucs, 1 );             
            }
            
            int maxx = 0; int maxy = 0;
            
            for( int x=0; x<ellista.Length+1; x++ ) {
                for( int y=0; y<ellista.Length+1; y++ ) {
                    if (max<d[ x, y ]) {
                        max = d[ x, y ];
                        maxx = x+1;
                        maxy = y+1;
                    }
                }   
            }
            
            Console.WriteLine( max );// + " : " + maxx + "-" + maxy );
            
        }
        
        private void beolvas() {
            StreamReader sr = new StreamReader("prufer.be");
            string s;
            ArrayList al = new ArrayList();     
            while( (s = sr.ReadLine()) != null ) {
                if( s != "0" ) {
                    string[] s2 = s.Split(' ');
                    int[] pr = new int[s2.Length-1];
                    for( int i=1; i<s2.Length; i++ )  {
                        pr[i-1] = Int32.Parse(s2[i]);
                    }
                    al.Add(pr);
                } else break;
            }
            sr.ReadLine();
            sr.Close();
                    
            prufers = al;
        }
        
        public static void Main(string[] args)
        {
            Program prg = new Program();
            prg.beolvas();
            for( int i=0; i<prufers.Count; i++ ) {
                prg.prufer = (int[])prufers[i];
                prg.doElLista();
                prg.doSzomszedsagiMatrix();
                prg.calcAtmero();
            }
            
            Console.ReadKey(true);
        }
    }
}
