program bacon;

var a,f,p :integer;
    aa:array[1..1000] of string;
    ff:array[1..2000] of string;
    pp:array[1..2000,1..2] of integer;
    x:array[1..3000,1..3000] of byte;
    sz1,sz2 :string;


procedure beolvas;
var t :text;
    i :integer;
begin
    assign(t,'actor.be');
    reset(t);
    readln(t,a);
    for i:=1 to a do
    begin
        readln(t,aa[i]);
    end;
    close(t);
    assign(t,'film.be');
    reset(t);
    readln(t,f);
    for i:=1 to f do
    begin
        readln(t,ff[i]);
    end;
    close(t);
    assign(t,'play.be');
    reset(t);
    readln(t,p);
    for i:=1 to p do
    begin
        readln(t,pp[i,1],pp[i,2]);
    end;
    close(t);
end;


procedure graf;
var i,j :integer;
begin
    for i:=1 to a do
    begin
        for j:=1 to f do
        begin
            x[i,j]:=0;
        end;
    end;
    for i:=1 to p do
    begin
        x[pp[i,1],pp[i,2]+a]:=1;
        x[pp[i,2]+a,pp[i,1]]:=1;
    end;
end;

function sorszam(k :string) :integer;
var i:integer;
begin
    for i:=1 to a do
    begin
        if aa[i]=k then sorszam:=i;
    end;
end;



procedure ut(kk,ll :string);
var k,l,i,j,q,o,r :integer;
    s:array[1..3000] of integer;
    v:array[1..3000,1..2] of integer;
    ki:array[1..1000] of integer;
begin
    o:=0;
    k:=sorszam(kk);
    l:=sorszam(ll);
    i:=1;
    j:=1;
    s[1]:=k;
    for q:=1 to (a+f) do
    begin
        v[q,1]:=-1;
    end;
    v[k,1]:=0;
    v[k,2]:=k;
    while (i<=j) and (o=0) do
    begin
        for q:=1 to (a+f) do
        begin
            if (x[s[i],q]=1) and (v[q,1]=-1) then
            begin
                v[q,1]:=v[s[i],1]+1;
                v[q,2]:=s[i];
                inc(j);
                s[j]:=q;
                if q=l then o:=1;
            end;
         end;
         inc(i);
    end;
    r:=l;
    writeln(v[l,1] div 2);
    for q:=v[l,1] downto 1 do
    begin
        ki[q]:=r;
        r:=v[r,2];
    end;
    writeln;
    write(aa[k],'->');
    for q:=1 to v[l,1]-1 do
    begin
        if q mod 2 = 0 then write(aa[ki[q]],'->');
        if q mod 2 = 1 then write(ff[ki[q]-a],'->');
    end;
    write(aa[l]);
end;




begin
beolvas;
graf;
writeln('Egyik színész: ');
readln(sz1);
writeln('Másik színész: ');
readln(sz2);
ut(sz1,sz2);
readln;
end.
