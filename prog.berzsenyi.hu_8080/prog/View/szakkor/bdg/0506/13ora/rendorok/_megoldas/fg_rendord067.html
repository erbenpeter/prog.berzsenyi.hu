{$R+} {$Q+}
PROGRAM Rendor2b;
CONST
    inputFile = '';
    maxn = 100;

VAR
    N: INTEGER;

    NO: BYTE;
    orsok: ARRAY [1..maxn] OF RECORD
        size: BYTE; {ennyi rendor van ezen az orson}
        pos: BYTE; {az ors pozicioja kezdetben az autopalyan}
        maxpos: BYTE; {az ors rendorei altal alkotott blokk
                        maximalis lehetseges kezdopozicioja (a megoldasban)}
    END;

    {A MinKoltseg fuggveny visszateresi ertekeit, es donteseit tarolja}
    cache: ARRAY [1..maxn, 1..maxn] OF RECORD
        koltseg, valasztas: INTEGER;
    END;

PROCEDURE Load;
VAR
    T: Text;
    i, j: BYTE;
BEGIN
    Assign(T, inputFile);
    Reset(T);
    ReadLn(T, N); NO:= 0;
    FOR i:= 1 TO N DO
    BEGIN
        Read(T, j);
        IF (j > 0) THEN
        BEGIN
            INC(NO);
            orsok[NO].size:= j; orsok[NO].pos:= i;
        END;
    END;
    Close(T);

    {a maxpos az utolso olyan pozicio, ahova meg leteheto az ors
    altal alkotott blokk eleje...}
    orsok[NO].maxpos:= N - orsok[NO].size + 1;
    FOR i:= NO - 1 DOWNTO 1 DO
    BEGIN
        orsok[i].maxpos:= orsok[i+1].maxpos - orsok[i].size;
    END;
END; {Load}

FUNCTION AthelyezesKoltseg(ors, uj_hely: INTEGER): INTEGER;
VAR
    i: INTEGER;
    k: INTEGER;
BEGIN
    k:= 0;
    FOR i:= uj_hely TO uj_hely+orsok[ors].size - 1 DO
    BEGIN
        k:= k + ABS( orsok[ors].pos - i );
    END;
    AthelyezesKoltseg:= k;
END; {AthelyezesKoltseg}

FUNCTION Koltseg(elso_ors, elso_hely: INTEGER): INTEGER;
VAR
    i, local, min, mini: INTEGER;
BEGIN
    IF (elso_ors > NO) THEN
    BEGIN
        Koltseg:= 0;
        EXIT;
    END;

    IF cache[elso_ors, elso_hely].koltseg > -2 THEN
    BEGIN
        Koltseg:= cache[elso_ors, elso_hely].koltseg;
        EXIT;
    END;

    min:= -1; mini:= -1;
    FOR i:= elso_hely TO orsok[elso_ors].maxpos DO
    BEGIN
        local:= Koltseg(elso_ors+1, i+orsok[elso_ors].size)
                + AthelyezesKoltseg(elso_ors, i);
            {Koltseg(..) sosem lehet -1, ha teljesul a "rendorok szama nemnagyobb, mint
             a helyek szama, mert a orsok[i].maxpos mindig biztositja, hogy csak
             megoldhato reszproblemara hivojon meg a fuggveny.}
        IF ((local < min) OR (min = -1)) THEN
        BEGIN
            min:= local;
            mini:= i;
        END;
    END;
    cache[elso_ors, elso_hely].koltseg:= min;
    cache[elso_ors, elso_hely].valasztas:= mini;
    Koltseg:= min;
END; {Koltseg}

PROCEDURE Megold;
VAR
    ors, hely, i: INTEGER;
    varosok: ARRAY [1..maxn] OF BYTE;
BEGIN
    {cache torles}
    FOR hely:= 1 TO maxn DO
    BEGIN
        FOR ors:= 1 TO maxn DO
        BEGIN
            cache[ors, hely].koltseg:= -2;
            cache[ors, hely].valasztas:= 0;
        END;
    END;

    {kiszamolas}
    WriteLn(Koltseg(1, 1));

    {optimalis megoldas kiirasa, eloszor a kovetkezo tombot toltjuk ki:}
    FOR i:= 1 TO maxn DO varosok[i]:= 0;

    {sorban vegigvesszuk a rendororsoket:}
    hely:= 1; ors:= 1;
    WHILE (ors <= NO) DO
    BEGIN
        hely:= cache[ors, hely].valasztas;
        FOR i:= hely TO hely + orsok[ors].size - 1 DO varosok[i]:= 1;
        hely:= hely + orsok[ors].size;
        ors:= ors + 1;
    END;
    FOR i:= 1 TO N DO
    BEGIN
        Write(varosok[i], ' ');
    END;
END; {Megold}

BEGIN
    Load;
    Megold;
END.
