using System;

namespace Abcode
{
    public class Coder
    {
        private string Beolvas()
        {
            return Console.ReadLine().ToUpper();
        }

        private void Kiir(string kod)
        {
            Console.WriteLine(kod);
        }
        
        private void Kodol()
        {
            int chrA = Convert.ToInt32('A');
            string kod = "", szoveg = Beolvas();

            for (int i = 0; i < szoveg.Length; i++)
                if (szoveg[i] >= 'A' && szoveg[i] <= 'Z')
                    kod += (Convert.ToInt32(szoveg[i]) - chrA + 1).ToString();

            Kiir(kod);
        }

        public static void Main(string[] args)
        {
            Coder coder = new Coder();
            coder.Kodol();
        }
    }
}
