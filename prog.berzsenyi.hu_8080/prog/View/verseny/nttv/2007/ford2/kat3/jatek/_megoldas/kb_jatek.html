using System;
using System.IO;
using System.Collections.Generic;

namespace Jatek
{

    class Field
    {
        public int x, y; // tudjuk magunkról hogy hol vagyunk a táblán.
        
        private int pearls = 0; // itt tároljuk, hogy max. hány gyöngyünk lehet az adott mezőn.
        private int defPearls; // itt tároljuk, hogy mennyi gyöngyöt gyűjthetünk az adott mezőn.
        private Field prevField = null; // itt tároljuk, hogy ide melyik mezőről jutottunk el.
        private bool done = false;
        
        public bool IsTrap {
            get {
                return defPearls == -1;
            }
        }

        public Field( int x, int y, int v )
        {
            this.x = x;
            this.y = y;
            this.defPearls = v;
            if( x == 0 && y == 0 ) {
                // ha az elsőn vagyunk, akkor az ottani gyöngyöket rögtön felszedjük.
                pearls = defPearls;
            }
        }

        public bool StepTo( Field to )
        {
            if( to.Pearls <= (this.Pearls + to.defPearls) ) {
                // ha megéri rálépni, akkor tegyük meg.
                to.Pearls = this.Pearls + to.defPearls;
                to.PreviousField = this; // elmentjük, hogy ide honnan célszerű érkezni.
                return true;
            }
            return false;
        }
        
        public Field PreviousField {
            get { return prevField; }
            set { prevField = value; }
        }
        
        public int Pearls {
            get { return pearls; }
            set { pearls = value; }
        }

        public bool IsDone {
            get { return done; }
            set { done = value; }
        }
        
    }
    
    class Board
    {
        private int n, m;
        private Field[,] fields;
        
        public Board( int n, int m, StreamReader ins )
        {
            this.n = n;
            this.m = m;
            fields = new Field[n,m];

            string[] line;
            for( int i=0; i<n; i++ ) {
                line = ins.ReadLine().Split(' ');
                for( int j=0; j<m; j++ ) {
                    fields[i, j] = new Field( j, i, int.Parse(line[j]) );
                }
            }
        }

        public Field GetField( int j, int i )
        {
            return fields[i, j];
        }

        public Field StepTo( Field from, int dx, int dy )
        {
            if( from.x + dx < m && from.y + dy < n ) {
                // ha létezik a mező, akkor megpróbálunk innen rálépni
                Field f = GetField( from.x+dx, from.y+dy );
                if( !f.IsTrap && !f.IsDone && from.StepTo( f ) ) {
                    return f; // ha léphetünk és megéri, akkor visszatérünk a mezővel
                }
            }
            return null; // különben nem.
        }

        public int Width { get{ return m; } }
        public int Height { get{ return n; } }
    }
    
    class Program
    {

        private Board board;
        
        public Program()
        {
            GetBoard(); // beolvassuk a táblát
            WalkOnTheBoard(); // végig lépkedünk a táblán a szabályoknak megfelelően
            GetSteps(); // visszakeressük a lépéseket
        }

        private void GetBoard()
        {
            StreamReader ins = new StreamReader("jatek.be");
            string[] line = ins.ReadLine().Split(' ');
            board = new Board( int.Parse(line[0]), int.Parse(line[1]), ins );
            ins.Close();
        }

        private void WalkOnTheBoard()
        {
            Queue<Field> q = new Queue<Field>(board.Width * board.Height);
            q.Enqueue( board.GetField(0, 0) );

            Field f, f2;
            while( q.Count != 0 ) {
                // lekérjük az adott mezőt és megnézzük a szomszédjait.
                f = q.Dequeue();
                // ha véletlenül mire ide jutunk, ő már bekerült a jó halmazba, akkor ne számolgassunk.
                // egyszerűen menjünk tovább
                if( f.IsDone ) continue;
                f.IsDone = true; // dijkstra szerint ő már minimális

                if( (f2 = board.StepTo(f, 1, 0)) != null ) {
                    q.Enqueue(f2);
                }
                if( (f2 = board.StepTo(f, 0, 1)) != null ) {
                    q.Enqueue(f2);
                }
            }
        }

        private void GetSteps()
        {
            StreamWriter outs = new StreamWriter("jatek.ki");
            Field f = board.GetField(board.Width-1, board.Height-1);
            
            if( f.PreviousField == null ) {
                outs.WriteLine(-1);
                outs.Close();
                return;
            }
            
            outs.WriteLine(f.Pearls);

            // nincs más dolgunk, mint lekérni az adott mező előtti mezőt, ezt pedig tudjuk!
            char[] steps = new char[board.Width+board.Height-2];
            int i = steps.Length; // hátulról töltjük fel!
            while( f.x != 0 || f.y != 0 ) {
                if( f.PreviousField.x < f.x ) {
                    steps[--i] = 'J'; // balról jöttünk ide, magyarán valójában jobbra léptünk
                } else if( f.PreviousField.y < f.y ) {
                    steps[--i] = 'L';
                }
                
                f = f.PreviousField;
            }

            for( i=0; i<steps.Length; i++ )
                outs.Write(steps[i]);
            outs.WriteLine();
            outs.Close();
        }
        
        public static void Main(string[] args)
        {
            new Program();
        }
    }
}
