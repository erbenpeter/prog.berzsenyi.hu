package minimalspanningtree;

/**
 *
 * @author 03cKrivan
 */
public class Case {

    public Edge[] edges;
    public int[] daddies; // apucikák
    public int points;
    
    public Case( int edges_num, int points_num )
    {
        edges = new Edge[edges_num];
        this.points = points_num;
    }

    private boolean areDifferentComponentsInEdge(int i) {
        int cOfP1 = getComponentOf( edges[i].getP1() );
        int cOfP2 = getComponentOf( edges[i].getP2() );
                        
        return (cOfP1 != cOfP2);
    }

    private int getComponentOf(int p) {
        while( daddies[p] >= 0 ) {
            p = daddies[p];
        }
        return p;
    }

    private int lengthOfTree( Tree edgeOfTree ) {
        Edge[] tree = (Edge[]) edgeOfTree.toArray( new Edge[ edgeOfTree.size() ] );
        int res = 0;
        for( int i=0; i<tree.length; i++ )
        {
            res += tree[i].getCost();
        }
        return res;
    }

    private Tree makeTree() {
        setDaddies();

        Tree resTree = new Tree();        
        for( int i=0; i<edges.length; i++ )
        {
            if( edges[i].isDisabled() ) continue;
            if( areDifferentComponentsInEdge( i ) )
            {
                resTree.add( edges[i] );                
                unionOfComponents( getComponentOf(edges[i].getP1()), getComponentOf(edges[i].getP2()) );
            }
        }
        return resTree;
    }

    private void setDaddies() {
        // egyszerűen az összes apuci => -1
        // ez reprezentálja azt, hogy mindenki külön komponentecske.
        daddies = new int[ points ];
        for(int i=0; i<daddies.length; i++)
        {
            daddies[i] = -1;
        }
    }
    
    private void sortEdges()
    {
        // ide jöhetne valami szép kis gyors rendezés :) most maradunk a min. kiválaszt.
        int minidx;
        for( int i=0; i<edges.length; i++ )
        {
            minidx = i;
            for( int m=i+1; m<edges.length; m++ )
            {
                if( edges[minidx].getCost() > edges[m].getCost() )
                {
                   minidx = m;
                }
            }
            // minidx = legkisebb. cseréljük le őt az i-re.
            int save_p1 = edges[i].getP1();
            int save_p2 = edges[i].getP2();
            int save_cost = edges[i].getCost();
            edges[i].setP1( edges[minidx].getP1() );
            edges[i].setP2( edges[minidx].getP2() );
            edges[i].setCost( edges[minidx].getCost() );
            edges[minidx].setP1( save_p1 );
            edges[minidx].setP2( save_p2 );
            edges[minidx].setCost( save_cost );
            // comment: uhh, de undorító.
        }
    }
    
    public void solve()
    {
        // megvannak az élek.
        // rendezzük őket.
        
        sortEdges(); // megvannak cost alapján NÖVEKVŐ sorrendben.
                
        Tree edgesOfOptTree = makeTree();
        int opt = lengthOfTree( edgesOfOptTree );        
                
        // most az edgesOfTree-ből tiltogatunk le éleket.
        int secondOpt = Integer.MAX_VALUE;
        for( int i=0; i<edgesOfOptTree.size(); i++ )
        {
            Edge tiltani = edgesOfOptTree.get( i );
            tiltani.disable();
            
            // most megint felépítjük a szitut.
            Tree edgesOfNonOptTree = makeTree();
            
            int potentialSecondOpt = lengthOfTree( edgesOfNonOptTree );
            if( secondOpt >= potentialSecondOpt )
            {
                secondOpt = potentialSecondOpt;
            }
            
            tiltani.enable();
        }

        System.out.println(opt + " " + secondOpt);
    }

    private void unionOfComponents(int c1, int c2) {
        // van két szaros componentünk
        // uniózzuk őket össze.
        if( -daddies[c1] > -daddies[c2] )
        {
            daddies[c1] += daddies[c2];
            daddies[c2] = c1;
        } else {
            daddies[c2] += daddies[c1];
            daddies[c1] = c2;
        }
    }
    
}

class Edge {

    private int p1, p2, cost;
    private boolean disabled = false;
    
    public Edge( int p1, int p2, int cost )
    {
        this.p1 = p1;
        this.p2 = p2;
        this.cost = cost;
    }
    
    public void setP1( int p )
    {
        this.p1 = p;
    }

    public void setP2( int p )
    {
        this.p2 = p;
    }
    
    public void setCost( int c )
    {
        this.cost = c;
    }
    
    public int getP1() { return this.p1; }
    public int getP2() { return this.p2; }
    public int getCost() { return this.cost; }
    
    @Override
    public Edge clone()
    {
        Edge res = new Edge( this.p1, this.p2, this.cost );
        return res;
    }
    
    public void disable()
    {
        this.disabled = true;
    }
    
    public void enable()
    {
        this.disabled = false;
    }
    
    public boolean isDisabled()
    {
        return (this.disabled);
    }

}
