float calcheight(int* r)
{
    float h=r[ro[0]];
    for(int i=0; i<n-1; i++)
    {
        h+=10*sqrt(2*r[ro[i]]+2*r[ro[i+1]]-100);
    }
    h+=r[ro[n-1]];

    return h;
}

bool compare(int p1, int p2)
{
    return p1<p2;
}

void next_order()
{
    bool done=false;
    for(int i=n-2; i>=0 && !done; i--)
    {
        if(ro[i+1]>ro[i])
        {
            int ind=i+1;
            for(int j=i+1; j<=n-1; j++)
                if(ro[j]>ro[i] && ro[j]<ro[ind])
                    ind=j;

            int v=ro[ind];
            ro[ind]=0;
            sort(ro+i, ro+n, compare);
            ro[i]=v;

            done=true;
        }
    }
}
