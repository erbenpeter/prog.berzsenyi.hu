using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

namespace Pakol
{
    class Thing : IComparable<Thing>
    {
        public int weight, id;

        public Thing(int _weight, int _id)
        {
            weight = _weight;
            id = _id;
        }

        public int CompareTo(Thing other)
        {
            return weight.CompareTo(other.weight);
        }
    }

    class Program
    {
        private List<Thing> things;
        private int[,] cache;
        private List<int>[,] choice;
        private int n, k;

        public Program()
        {
            StreamReader ins = new StreamReader("pakol.be");
            string[] line = ins.ReadLine().Split(' ');
            n = int.Parse(line[0]);
            k = int.Parse(line[1]);
            cache = new int[n, k+1];
            choice = new List<int>[n, k+1];

            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < k+1; j++)
                {
                    cache[i, j] = -1;
                    choice[i, j] = new List<int>();
                }
            }

            line = ins.ReadLine().Split(' ');
            things = new List<Thing>(n);
            for (int i = 0; i < n; i++)
            {
                things.Add(new Thing(int.Parse(line[i]), i + 1));
            }

            things.Sort();

            StreamWriter outs = new StreamWriter("pakol.ki");
            outs.WriteLine(k-check( 0, k )); // 0-tól indulunk, még k hely van!
            
            List<int> solution = choice[0, k];
            outs.WriteLine(solution.Count);
            for (int i = 0; i < solution.Count; i++)
            {
                outs.Write( solution[i] + " " );
            }
            outs.WriteLine();

            outs.Close();
        }

        private int check(int from, int rest)
        {
            int min = 0;
            if (from == n) return rest;
            if (cache[from, rest] == -1)
            {
                if (things[from].weight <= rest)
                {
                    // belefér!
                    int a = check(from + 1, rest - things[from].weight);
                    int b = check(from + 1, rest);
                    if (a < b)
                    {
                        // a betevős az jobb, akkor megjegyezzük, hogy inkább betettük
                        if (from + 1 != n)
                        {
                            choice[from, rest].AddRange(choice[from + 1, rest - things[from].weight]);
                        }
                        choice[from, rest].Add(things[from].id);
                    }
                    else
                    {
                        if (from + 1 != n)
                            choice[from, rest].AddRange(choice[from + 1, rest]);
                    }

                    cache[from, rest] = Math.Min(a, b);
                }
                else
                {
                    // nem fér bele nézzük a következőt.
                    cache[from, rest] = rest;
                }
            }

            return cache[from, rest];
        }

        static void Main(string[] args)
        {
            new Program();
        }
    }
}
