using System.Collections.Generic;
using System.Text;

namespace Tili_Toli

    class Program
    {
        const int fs = 50;
        static void Main(string[] args)
        {
            ToliGame game = new ToliGame(4, 5,
                new ToliElem[10] { 
                    new ToliElem(1, 2, 0, 0), new ToliElem(1, 2, 3, 0), new ToliElem(1, 2, 0, 3), new ToliElem(1, 2, 3, 3), //fekete
                    new ToliElem(1, 1, 1, 3), new ToliElem(1, 1, 2, 3), new ToliElem(1, 1, 1, 4), new ToliElem(1, 1, 2, 4), //piros
                    new ToliElem(2, 1, 1, 2), new ToliElem(2, 2, 1, 0)
                },
                new ToliElem(2, 2, 1, 3));

            List<ulong> s = game.Solve();
            SolidBrush[] col = new SolidBrush[4] { new SolidBrush(Color.Blue), new SolidBrush(Color.Yellow), new SolidBrush(Color.Red), new SolidBrush(Color.DeepPink) };

            for (int i = 0; i != s.Count; i++)
            {
                Bitmap b = new Bitmap(4 * fs, 5 * fs, System.Drawing.Imaging.PixelFormat.Format24bppRgb);
                Graphics g = Graphics.FromImage(b);
                g.FillRectangle(Brushes.White, 0, 0, 4 * fs, 5 * fs);
                ToliGame.ToliE[] d = game.Decode(s[i]);
                int c = -1;

                for (int j = 0; j != d.Length; j++)
                {

                    if (j == 0 || d[j - 1].x != d[j].x || d[j - 1].y != d[j].y) c++;

                    g.FillRectangle(col[c], d[j].px * fs, d[j].py * fs, d[j].x * fs-1, d[j].y * fs-1);
                    g.DrawRectangle(Pens.Black, d[j].px * fs, d[j].py * fs, d[j].x * fs-1, d[j].y * fs-1);
                }
                int name=s.Count-i;
                b.Save("lepes" + name.ToString("D3") + ".bmp");
            }
        }

        public struct ToliElem : IComparable<ToliElem>
        {
            public byte x, y, px, py; //méretek, hely.

            public ToliElem(byte xi, byte yi, byte pxi, byte pyi)
            {
                x = xi; y = yi; px = pxi; py = pyi;
            }

            public int CompareTo(ToliElem other)
            {
                if (other.x > x) return -1;
                if (other.x < x) return 1;
                if (other.y > y) return -1;
                if (other.y < y) return 1;
                return 0;
            }
        }

        public class ToliGame
        {
            byte tx, ty;        //tábla mérete
            ToliElemT[] te;     //toli elem típusok
            int n;              //elemek száma
            ulong State0;       //kiinduló állapot
            ToliElem e;         //végállapot
            int EM;             //melyik elemet kell végállapottal hasonlítani
            int[] ord;          //rendező tömb


            public struct StatePair : IComparable<StatePair>
            {
                public ulong Child, Parent;

                public int CompareTo(StatePair other)
                {
                    if (other.Child > Child) return -1;
                    if (other.Child == Child) return 0;
                    return 1;
                }

                public StatePair(ulong c, ulong p)
                {
                    Child = c;
                    Parent = p;
                }
            }

            public List<ulong> Solve()
            {
                List<StatePair> u = new List<StatePair>(); //ami már megvan
                LinkedList<StatePair> sor = new LinkedList<StatePair>(); //ez lesz itt most hatékony
                sor.AddLast(new StatePair(State0, 0));

                StatePair Solution = new StatePair(0, 0);
                while (sor.Count > 0)
                {
                    StatePair s = sor.First.Value; sor.RemoveFirst();
                    if (u.BinarySearch(s) >= 0)
                        continue;
                    u.Add(s);
                    u.Sort();

                    int v = -1;
                    List<StatePair> g = Generate(s.Child, ref v);
                    if (v != -1) { Solution = g[v]; break; }
                    for (int i = 0; i != g.Count; i++)
                    {
                        if (u.BinarySearch(g[i]) < 0) //ha még nem szerepel..
                        {
                            sor.AddLast(new LinkedListNode<StatePair>(g[i]));
                        }

                    }
                }

                List<ulong> l = new List<ulong>();
                if (Solution.Child == 0) return l;  //nincs megoldás

                while (true)
                {
                    l.Add(Solution.Child);
                    ToliE[] d = Decode(Solution.Child);
                    if (Solution.Parent == 0) break;
                    int k = u.BinarySearch(new StatePair(Solution.Parent, 0));
                    Solution = u[k];
                }

                return l;
            }

            private List<StatePair> Generate(ulong p, ref int v)
            {
                ToliE[] elemek = Decode(p);

                bool[,] foglalt = new bool[tx, ty];
                for (int i = 0; i != n; i++)
                    for (int j = 0; j != elemek[i].x; j++)
                        for (int k = 0; k != elemek[i].y; k++)
                            foglalt[j + elemek[i].px, k + elemek[i].py] = true;

                List<StatePair> sp = new List<StatePair>();

                for (int i = 0; i != n; i++)
                {
                    for (int j = 0; j != elemek[i].x; j++)
                        for (int k = 0; k != elemek[i].y; k++)
                            foglalt[j + elemek[i].px, k + elemek[i].py] = false;

                    elemek[i].px--;
                    if (Ellenoriz(elemek[i], foglalt))
                    {
                        sp.Add(new StatePair(Encode(elemek), p)); //balra
                        if (elemek[EM].px == e.px && elemek[EM].py == e.py)
                        { v = sp.Count - 1; return sp; }
                    }

                    elemek[i].px++; elemek[i].px++;
                    if (Ellenoriz(elemek[i], foglalt))
                    {
                        sp.Add(new StatePair(Encode(elemek), p)); //jobbra
                        if (elemek[EM].px == e.px && elemek[EM].py == e.py)
                        { v = sp.Count - 1; return sp; }
                    }

                    elemek[i].px--; elemek[i].py--;
                    if (Ellenoriz(elemek[i], foglalt))
                    {
                        sp.Add(new StatePair(Encode(elemek), p)); //fönt
                        if (elemek[EM].px == e.px && elemek[EM].py == e.py)
                        { v = sp.Count - 1; return sp; }
                    }

                    elemek[i].py++; elemek[i].py++;
                    if (Ellenoriz(elemek[i], foglalt))
                    {
                        sp.Add(new StatePair(Encode(elemek), p)); //lent
                        if (elemek[EM].px == e.px && elemek[EM].py == e.py)
                        { v = sp.Count - 1; return sp; }
                    }

                    elemek[i].py--;

                    for (int j = 0; j != elemek[i].x; j++)
                        for (int k = 0; k != elemek[i].y; k++)
                            foglalt[j + elemek[i].px, k + elemek[i].py] = true;
                }

                return sp;
            }

            private bool Ellenoriz(ToliE elem, bool[,] foglalt)
            {
                int a, b;
                for (int j = 0; j != elem.x; j++)
                {
                    for (int k = 0; k != elem.y; k++)
                    {
                        a = j + elem.px;
                        b = k + elem.py;
                        if (a < 0 || b < 0 || a + 1 > tx || b + 1 > ty) return false;
                        if (foglalt[a, b]) return false;
                    }
                }
                return true;
            }

            #region Készen levő eljárások

            private byte ReqBits(uint var) //hányféle lehetőséget kell tárolni
            {
                byte i = 0;
                while (true)
                {
                    if (var <= (1 << i)) break;
                    i++;
                }
                return i;
            }

            public struct ToliE : IComparable<ToliE>
            {
                public byte px, py, bx, by, x, y;

                public ToliE(byte x, byte y, byte px, byte py, byte bx, byte by)
                {
                    this.px = px; this.py = py; this.bx = bx; this.by = by; this.x = x; this.y = y;
                }

                public int CompareTo(ToliE other)
                {
                    if (other.x > x) return -1;
                    if (other.x < x) return 1;
                    if (other.y > y) return -1;
                    if (other.y < y) return 1;
                    if (other.px > px) return -1;
                    if (other.px < px) return 1;
                    if (other.py > py) return -1;
                    if (other.py < py) return 1;
                    return 0;
                }
            }

            private struct ToliElemT
            {
                public byte x, y; //méretek
                public byte no; //hány darab
                public byte bx, by; //hány bit kell egy elem tárolásához

                public ToliElemT(byte x, byte y, byte no, byte bitsx, byte bitsy)
                {
                    this.x = x;
                    this.y = y;
                    this.no = no;
                    this.bx = bitsx;
                    this.by = bitsy;
                }
            }

            public ulong Encode(ToliE[] t)
            {
                Array.Sort(t, ord);

                ulong State = 0;
                for (int i = 0; i != n; i++)
                {
                    State = State << t[i].bx;
                    State += t[i].px;
                    State = State << t[i].by;
                    State += t[i].py;
                }

                Array.Sort(ord, t);

                return State;
            }

            public ToliE[] Decode(ulong State)
            {
                ToliE[] t = new ToliE[n];
                int c = n;

                for (int i = te.Length - 1; i != -1; i--)
                {
                    for (int j = 0; j != te[i].no; j++)
                    {
                        c--;
                        t[c].x = te[i].x; t[c].y = te[i].y; t[c].bx = te[i].bx; t[c].by = te[i].by;

                        t[c].py = (byte)((State << (64 - t[c].by)) >> (64 - t[c].by));
                        State = State >> t[c].by;
                        t[c].px = (byte)((State << (64 - t[c].bx)) >> (64 - t[c].bx));
                        State = State >> t[c].bx;
                    }
                }

                return t;
            }

            public ToliGame(byte x, byte y, ToliElem[] t, ToliElem EndState) //tábla mérete, elemek, végállapot
            {
                tx = x;
                ty = y;
                n = t.Length;
                e = EndState;

                Array.Sort(t);
                List<ToliElemT> r = new List<ToliElemT>();
                ToliE[] q = new ToliE[n];

                int bits = 0;

                for (int i = 0; i != n; i++)
                {
                    if (r.Count == 0 || r[r.Count - 1].x != t[i].x || r[r.Count - 1].y != t[i].y)
                    {
                        //új elem szükségeltetik
                        if (t[i].x > tx || t[i].y > ty) throw new Exception("Túl nagy elem");
                        r.Add(new ToliElemT(t[i].x, t[i].y, 1, ReqBits((uint)(tx + 1 - t[i].x)), ReqBits((uint)(ty + 1 - t[i].y))));
                    }
                    else
                    {
                        ToliElemT z = r[r.Count - 1];
                        z.no++;
                        r[r.Count - 1] = z;
                    }

                    q[i] = new ToliE(t[i].x, t[i].y, t[i].px, t[i].py, r[r.Count - 1].bx, r[r.Count - 1].by);
                    try { bits += r[r.Count - 1].bx + r[r.Count - 1].by; }
                    catch { throw new Exception("64 bit nem elég egy állapot tárolásához"); }

                    if (e.x == t[i].x && e.y == t[i].y) EM = i;
                }

                if (bits > 64) throw new Exception("64 bit nem elég egy állapot tárolásához");

                te = r.ToArray();

                ord = new int[n];
                for (int i = 0; i != n; i++) ord[i] = i;

                State0 = Encode(q);

            }

            #endregion

        }
    }
