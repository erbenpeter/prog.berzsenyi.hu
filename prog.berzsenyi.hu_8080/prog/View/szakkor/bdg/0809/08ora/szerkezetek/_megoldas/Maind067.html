package hu.krivan.szakkor.szerkezetek;

import java.io.BufferedWriter;
import java.io.FileNotFoundException;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter;

/**
 *
 * @author Bálint Kriván
 */
public class Main {

    private static final String MAGIC_STRING = "impossible";

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) throws FileNotFoundException, IOException {
        PrintWriter out = new PrintWriter( new BufferedWriter( new FileWriter("myOutput.txt") ) );

        Graf graf;
        while( (graf = GrafFactory.getNextGraf()) != null )
        {
            int height = graf.maxHeightIfHung();
            if( height == -1 ) {
                out.println( MAGIC_STRING );
            } else {
                out.println( height );
            }
        }

        out.close();
    }

}
