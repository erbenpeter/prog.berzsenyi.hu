﻿using System;

namespace hatodik
{
    class particio
    {
        public int[] t;
        public int c = 0;
        public void par(int n,int m)
        {
            if (n == 0)
            {   
                for (int z = 1;z <= m;z++)
                {
                    Console.Write("{0} ",t[z]);
                }
                Console.WriteLine();
                c++;
            }   else 
            {
                for (int i = t[m];i <= n;i++)
                {
                    t[m+1] = i;
                    par(n - i,m + 1);
                }
            }
            /*
            par(n,m)
                Ha n = 0 akkor az a[1],a[2],...,a[m] partíció bejárása
                különben 
                    Ciklus i := a[m]-től n-ig
                        a[m+1] := i
                        par(n - i, m + 1)
                    Ciklus vége
                Elágazás vége       
            Eljárás vége
            
            Indítás:
                par(n,0)
             */
        }
        
        public static void Main()
        {
            particio p = new particio();
            int d = Int32.Parse(Console.ReadLine());
            p.t = new int[d+1];
            p.t[0] = 1;
            
            p.par(d,0);             
            Console.WriteLine(p.c);
            //Console.Write("Press any key to continue . . . ");
            Console.ReadKey(true);
        }
    }
}
