{$R+} {$Q+}
PROGRAM Virus1;
CONST
    {chr = '0';}
    inputFile = ''{'virus'+chr+'.be'};
    outputFile = ''{'virus'+chr+'.ki1'};
    {^^---- standard I/O}
    maxIdo = 7000;

VAR
    H: INTEGER;
    N: LONGINT;
    virusok: ARRAY [1..maxIdo] OF LONGINT;


PROCEDURE Init;
VAR
    i: INTEGER;
BEGIN
    FOR i:= 1 TO maxIdo DO
    BEGIN
        virusok[i] := 0;
    END;
END; {Init}

PROCEDURE Hozzaad(k, p: INTEGER); INLINE;
VAR
    i: INTEGER;
BEGIN
    k:= k-H+1;
    IF k < 1 THEN k:= 1;
    FOR i:= k TO p-1 DO
    BEGIN
        INC(virusok[i]);
    END;
END; {Hozzaad}

PROCEDURE Process;
VAR
    T: Text;
    i: LONGINT;
    k, p: INTEGER;
BEGIN
    Assign(T, inputFile);
    Reset(T);
    ReadLn(T, N, H);
    FOR i:= 1 TO N DO
    BEGIN
        ReadLn(T, k, p);
        Hozzaad(k, p);
    END;
    Close(T);
END; {Process}

PROCEDURE Kiir;
VAR
    maxV, maxT: LONGINT;
    i: INTEGER;
    T: Text;
BEGIN
    maxV := virusok[1]; maxT := 1;
    FOR i:= 2 TO maxIdo-H DO
    BEGIN
        IF virusok[i] > maxV THEN
        BEGIN
            maxT := i;
            maxV := virusok[i];
        END;
    END;
    Assign(T, outputFile);
    Rewrite(T);
    WriteLn(T, maxV, ' ', maxT);
    Close(T);
END; {Kiir}

BEGIN
    Init;
    Process;
    Kiir;
END.
