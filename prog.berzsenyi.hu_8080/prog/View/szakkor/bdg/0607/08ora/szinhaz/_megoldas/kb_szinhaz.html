using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

namespace Szinhaz
{
    class Order
    {
        public int x, y;
        public int length;
        public int id;

        public Order( int _x, int _y, int _id )
        {
            x = _x;
            y = _y;
            length = y - x + 1;
            id = _id;
        }
    }

    class Program
    {
        private int n, k;
        private List<Order> orders;
        private List<int[]> solutions;

        public Program()
        {
            StreamReader ins = new StreamReader("szinhaz9.be");
            StreamWriter outs = new StreamWriter("szinhaz9.ki");

            string[] line = ins.ReadLine().Split(' ');
            n = int.Parse(line[0]);
            k = int.Parse(line[1]);

            int id = 1;
            orders = new List<Order>(k);
            for (int i = 0; i < k; i++)
            {
                line = ins.ReadLine().Split(' ');
                orders.Add( new Order( int.Parse(line[0])-1, int.Parse(line[1])-1, id++ ) );
            }

            solutions = new List<int[]>();
            Order o;
            for (int i = 0; i < n; i++)
            {
                // megyünk sorban az ülőhelyeken.
                o = getAnOrder( i );
                if (o != null)
                {
                    solutions.Add(new int[] { o.id, i });
                }
            }

            outs.WriteLine(solutions.Count);
            foreach (int[] c in solutions)
            {
                outs.WriteLine( "{0} {1}", c[0], c[1] );
            }

            outs.Flush();
        }

        private Order getAnOrder(int i)
        {
            int min = n+1;
            Order best = null;
            foreach (Order o in orders)
            {
                if (o.x <= i && o.y >= i)
                {
                    if (min > o.length)
                    {
                        min = o.length;
                        best = o;
                    }
                }
            }
            if (best == null) return null;
            best.length = 1; // őt is vegyük ki, akármi lesz is vele!
            for (int j = 0; j < orders.Count; j++ )
            {
                if (orders[j].x <= i && orders[j].y >= i)
                {
                    orders[j].length--;
                    if (orders[j].length == 0)
                    {
                        orders.RemoveAt(j);
                        j--; // kivettünk 1-et!
                    }
                }
            }
            return best;
        }

        static void Main(string[] args)
        {
            new Program();
        }
    }
}
