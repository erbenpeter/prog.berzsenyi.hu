﻿using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;

namespace Idegen_nyelv
{
    class Program
    {
        public static int wl, ec, wc; //wl=szóhossz ec=létező szavak darabszáma //wc csekkolandó szavak darabszáma
        static string[] exwords;
        static string[] words;
        public static List<char>[,] wordtokens;
        static int[] results;
        static Fa fa;
        static void Beolvas(string file)
        {
            StreamReader sr = new StreamReader(file);
            string line = sr.ReadLine();
            string[] a3 = new string[3];
            a3 = line.Split(' ');
            wl = Convert.ToInt32(a3[0]);
            ec = Convert.ToInt32(a3[1]);
            wc = Convert.ToInt32(a3[2]);

            exwords = new string[ec];
            for (int i = 0; i != ec; i++)
            {
                exwords[i] = sr.ReadLine();
            }

            words = new string[wc];
            for (int i = 0; i != wc; i++)
            {
                words[i] = sr.ReadLine();
            }

            sr.Close();
        }

        static void CreateTokens()
        {
            wordtokens = new List<char>[wc, wl];
            for (int i = 0; i != wc; i++)
            {
                int addto = 0;
                char c;
                bool p = false;

                for (int j = 0; j != words[i].Length; j++)
                {
                    c = words[i][j];
                    if (c == '(') { p = true; continue; }
                    if (c == ')') { p = false; addto++; continue; }
                    if (wordtokens[i, addto] == null) wordtokens[i, addto] = new List<char>();
                    wordtokens[i, addto].Add(c);
                    if (!p) addto++;
                }
            }
        }

        static void Munka()
        {
            CreateTokens();
            GenerateSzofa();

            results = new int[wc];
            for (int i = 0; i != wc; i++)
                results[i] = fa.Fit(i);
            
        }

        static void GenerateSzofa()
        {
            fa = new Fa();
            foreach (string w in exwords) fa.Expand(w);
        }

        static void Kiir(string file)
        {
            StreamWriter sw = new StreamWriter(file);
            for (int i = 0; i != wc; i++)
            {
                sw.WriteLine("Case #"+(i+1)+": " + results[i]);
            }
            sw.Close();
        }

        static void Main(string[] args)
        {
            string input = "lang-large.in";
            string output = "lang-large.out";

            DateTime t1 = new DateTime();
            DateTime t2 = new DateTime();
            
            Beolvas(input);

            t1 = DateTime.Now;
            Munka();
            t2 = DateTime.Now;

            Kiir(output);
            
            Console.WriteLine("Done " + (t2-t1).TotalMilliseconds + " milliseconds");
            Console.Read();
        }
    }

    class Fa
    {
        public char c;
        public int n;
        public int lvl;
        Fa parent;
        Fa[] children;

        public int Fit(int id) //word id
        {
            if (lvl == Program.wl) return 1;

            int count = 0;
            foreach (char c in Program.wordtokens[id, lvl])
            {
                if (HasChild(c))
                {
                    count += GetChild(c).Fit(id);
                }
            }
            return count;

        }

        public Fa(Fa parent, char c)
        {
            this.parent = parent;
            this.c = c;
            this.n = CharToNum(c);
            this.lvl = parent.lvl + 1;
            children = new Fa[26];
            this.parent.AddChild(this);
        }

        public Fa()
        {
            children = new Fa[26];
            lvl = 0;
        }

        public void AddChild(Fa child)
        {
            this.children[child.n] = child;
        }

        public bool HasChild(char c)
        {
            if (children[CharToNum(c)] != null) return true;
            return false;
        }

        public Fa GetChild(char c)
        {
            return children[CharToNum(c)];
        }

        public void Expand(string s)
        {
            if (s == "") return;
            if (!HasChild(s[0]))
            {
                new Fa(this, s[0]);
            }
            GetChild(s[0]).Expand(s.Substring(1));
        }

        public override string ToString()
        {
            string re = c.ToString().ToUpper();
            foreach (Fa child in children)
                re += " " + child.c;
            return re;
        }

        static int CharToNum(char c)
        {
            string abc = "abcdefghijklmnopqrstuvwxyz";
            for (int i = 0; i != abc.Length; i++)
                if (c == abc[i]) return i;
            throw new Exception("Char c is not a letter.");
        }
    }
}
