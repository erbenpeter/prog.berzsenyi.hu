#include <stdio.h>

int main(int argc, char **argv)
{
    char b;

    while ((b = getchar()) != EOF)
        if (b >= 'a' && b <= 'z')
            printf("%d", b - 'a' + 1);
        else if (b >= 'A' && b <= 'Z')
            printf("%d", b - 'A' + 1);

    printf("\n");

    return 0;
}
