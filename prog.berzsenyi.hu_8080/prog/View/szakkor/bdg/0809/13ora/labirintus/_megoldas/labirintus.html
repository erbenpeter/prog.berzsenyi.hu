#include <stdio.h>
#include <conio.h>
#include <matrix.h>

#define MaxRLen 10000

cMatrix Lab[2], Switch;

char MinR[MaxRLen];
int MinRLen, bMin;

int Read (char *);
void Backtrack (void);
int Write (char *);

int main (int AN, char * Args []) {
    if (AN < 3)
        return 1;
    if (!Read(Args[1]))
        return 1;
    Backtrack();
    if (!Write(Args[2]))
        return 1;
    return 0;
}

int Read (char * FN) {
    FILE * F = fopen(FN, "rt");
    int N, M, I, J, A;
    
    if (F == NULL)
        return 0;
    // A feladathoz képpest fel van cserélve N és M a matrix.h miatt
    fscanf(F, "%d %d\n", &N, &M);
    Lab[0].Set(N, M);
    Lab[1].Set(N, M);
    Switch.Set(N, M);
    for (I = 0; I < N; I++)
        for (J = 0; J < M; J++) {
            fscanf(F, "%d ", &A);
            Lab[0].SetItem(J, I, A == 0 || A == 2 || A == 3);
            Lab[1].SetItem(J, I, A == 0 || A == 2 || A == 4);
            Switch.SetItem(J, I, A == 2);
        }
    fclose(F);
    return 1;
}
/*
    Backtrack.
    Lab[] lehetséges értékei:
        0: fal vagy csukott ajtó
        1: üres vagy nyitott ajtó
        2: innen jobbra léptünk
        3: innen lefelé léptünk
        4: innen balra léptünk
        5: innen felfelé léptünk
        Az első mátrix a forrásnak megfelelő, a második mátrix pedig a vele
        ellentétes ajtóállást tárolja.
    Switch: logikai mátrix a kapcsolók tárolására.
    X, Y: az éppen aktuális koordináta
    Z: az ajtók állapota (a Lab[] indexe)
    R, RLen: Az út leírása (0 - 3 irányokkal), hossza
    MinR, MinRLen : a minimális út leírása, hossza
    bMin: van-e már minimum elmentve
    A[][]: a négy irány koordinátái
*/
void Backtrack (void) {
    int I, X = 0, Y = 0, Z = 0, RLen = 0, H = 0;
    int A[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
    char R[MaxRLen];
    
    while (1) {
        // Ha célbaértünk, és még nem volt ilyen rövid út, akkor elmentjük
        if (X == Lab[Z].GetM() - 1 && Y == Lab[Z].GetN() - 1)
            if (!bMin || RLen < MinRLen) {
                for (I = 0; I < RLen; I++)
                    MinR[I] = R[I];
                MinRLen = RLen;
                bMin = 1;
            }
        // Ciklus a négy irányba
        for (I = Lab[Z].GetItem(X, Y) - 1; I < 4; I++)
            if (Lab[Z].GetItem(X + A[I][0], Y + A[I][1]) == 1) {
                Lab[Z].SetItem(X, Y, I + 2);
                X+= A[I][0];
                Y+= A[I][1];
                if (Switch.GetItem(X, Y))
                    Z = !Z;
                R[RLen++] = I;
                break;  // csak a for ciklusból lép ki
            }
        // Ha nem tudtunk sehova lépni, akkor visszalépünk, vagy kilépünk
        if (I == 4) {
            Lab[Z].SetItem(X, Y, 1);
            if (RLen == 0)
                break;
            RLen--;
            if (Switch.GetItem(X, Y))
                Z = !Z;
            X-= A[R[RLen]][0];
            Y-= A[R[RLen]][1];
        }
    }
}
int Write (char * FN) {
    FILE * F = fopen(FN, "wt");
    char L[4] = {'J', 'L', 'B', 'F'};
    
    if (F == NULL)
        return 0;
    if (bMin) {
        fprintf(F, "%d\n", MinRLen);
        for (int I = 0; I < MinRLen; I++)
            fprintf(F, "%c", L[MinR[I]]);
    } else
        fprintf(F, "0");
    fclose(F);
    return 1;
}
