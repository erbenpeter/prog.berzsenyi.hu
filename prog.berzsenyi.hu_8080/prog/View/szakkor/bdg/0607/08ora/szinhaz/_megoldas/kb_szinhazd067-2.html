uses crt, strings;

const fajl = 'szinhaz9.be';

var N, M : integer;
    rendelesek : array[1..1000,1..3] of integer;
    eredmeny : array[1..1000,1..2] of integer;
    db : integer;

procedure beolvas;
var f:text;
    i:integer;
begin
     assign(f, fajl);
     reset(f);
     readln(f, M, N);
     for i:=1 to N do begin
         readln(f, rendelesek[i][1], rendelesek[i][2] );
         rendelesek[i][3] := rendelesek[i][2]-rendelesek[i][1]+1;
     end;
     close(f);
end;

end;

procedure kivesz( hely:integer );
var i:integer;
begin
     for i:=1 to N do begin
         if( (hely >= rendelesek[i][1]) and (hely <= rendelesek[i][2]) and (rendelesek[i][3] <> 0) ) then begin
             dec( rendelesek[i][3] );
         end;
     end;
end;

procedure vizsgal;
var i:integer;
    idx:integer;
begin

     db := 0;

     for i:=1 to M do begin
         {vegig nezzuk az ulohelyeket}
         idx := keres(i);
         if( idx < N+1 ) then begin
         inc(db);
         {megvan, hogy melyik rendelesbol vesszuk az ulohelyet, vegyuk ki mindenhonnan ezt az ulohelyet, de eleg,
         hogyha a hosszat roviditjuk}
         rendelesek[idx][3] := 0;
         kivesz(i);
         eredmeny[db][1] := idx;
         eredmeny[db][2] := i;
         end;
     end;

end;

procedure kiir;
var f:text;
    i:integer;
begin
     assign(f, 'out.txt');
     rewrite(f);

     writeln(f, db);

     for i:=1 to db do begin
         writeln(f, eredmeny[i][1], ' ', eredmeny[i][2]);

     end;

     close(f);
end;
