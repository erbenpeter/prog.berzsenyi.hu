using System;
using System.IO;
using System.Collections.Generic;

namespace darab
{
       class Program
       {
               public struct Problem
               {
                       public int x; public int y; public int db;
                       public Problem(int x, int y, int db)
                       {
                               this.x=x;
                               this.y=y;
                               this.db=db;
                       }
               }

               public static List<Problem> problems=new List<Problem>();

               public static int[,,] cache;
               public static Problem akt;

               public static int munka(Problem p)
               {
                       akt=p;
                       cache=new int[p.x+1, p.y+1, p.db+1];
                       for (int i=1; i!=p.x+1; i++)
                       {
                               for (int j=1; j!=p.y+1; j++)
                               {
                                       cache[i, j, 1]=i*j;
                               }
                       }
                       return opt(p.x, p.y, p.db);
               }

               public static int opt(int x, int y, int db) //x*y-os téglalap, db részre osztva, mennyi a minimum
               {
                       if ((x==0) || (y==0) || (db==0))
                               return int.MaxValue; //ezt ki akarjuk zárni

                       if (cache[x, y, db]!=0)
                               return cache[x, y, db]; //már kiszámoltuk

                       //ha nem...

                       int min=int.MaxValue;
                       int ertek=int.MaxValue;
                       int m1;
                       int m2;


                       for (int c=1; c!=db; c++) //c a darabszám
                       {
                               for (int i=1; i!=x; i++) //egyik tengely
                               {
                                       m1=opt(i, y, c);
                                       m2=opt(x-i, y, db-c);
                                       ertek=Math.Max(m1, m2);
                                       if (ertek<min) min=ertek;
                               }
                               for (int j=1; j!=y; j++) //másik tengely
                               {
                                       m1=opt(x, j, c);
                                       m2=opt(x, y-j, db-c);
                                       ertek=Math.Max(m1, m2);
                                       if (ertek<min) min=ertek;
                               }
                       }
                       cache[x, y, db]=min;

                       return min;
               }

               public static void Main(string[] args)
               {
                       int[] megoldasok;
                       beolvas("darabol.in");
                       megoldasok=new int[problems.Count];
                       for (int i=0; i!=problems.Count; i++) megoldasok[i]=munka(problems[i]);
                       kiir("darabol.out", megoldasok);
               }

               public static void beolvas(string inp)
               {
                       TextReader tr=new StreamReader(inp);
                       string[] l;
                       while((l=tr.ReadLine().Split(' '))[0]!="0")
                       {
                               Problem p=new Problem(ti(l[0]), ti(l[1]), ti(l[2]));
                               problems.Add(p);
                       }
                       tr.Close();
               }

               public static int ti(string s)
               {
                       return Convert.ToInt32(s);
               }

               public static void kiir(string oup, int[] mok)
               {
                       TextWriter tw=new StreamWriter(oup);
                       for (int i=0; i!=mok.Length; i++) tw.WriteLine(mok[i]);
                       tw.Close();
               }
       }
}
