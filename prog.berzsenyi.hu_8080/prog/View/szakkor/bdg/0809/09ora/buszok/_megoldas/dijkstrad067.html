#define type unsigned long
#define inf -1L
#define MaxNum 10
#define MaxBusN 50

#include <stdio.h>
#include "matrix.h"

/*
    Dijkstra algoritmus alkalmazása
    A programot Uray M. János készítette 2008. november 18-án.
*/

// összes pont adata az egyikre nézve a Dijkstra algoritmushoz
struct {
    // távolság
    type D;
    // kék-e, előző sorszáma
    int bDone, Prev;
} S[MaxNum];
// gráf, és éleinek száma
cMatrix Graph;
int Num;
struct point {
    type X, Y;
};

// összeadás a végtelennel kiterjesztve
inline type add (type A, type B) {
    if (A == inf || B == inf)
        return inf;
    return A + B;
}
inline long abs (long A) {
    return A < 0 ? -A : A;
}
// két pont távolsága
inline type Dist (point A, point B) {
    return abs(A.X - B.X) + abs(A.Y - B.Y);
}

type Dist (point, point, point);
type Dist (point, point, point, point);
void Dijkstra (int);

int main (int AN, char * Args []) {
    int I, J, K, L, R;
    type D, Min;
    point P[2];
    FILE * IF, * OF;
    struct sBus {
        int N, Prise;
        point P[MaxBusN];
    } * Bus;

    printf("<-- Start -->\n");

    if (AN < 3)
        return 1;
    IF = fopen(Args[1], "rt");
    if (IF == NULL)
        return 1;
    fscanf(IF, "%ld\n", &D);
    fscanf(IF, "%ld %ld\n", &P[0].X, &P[0].Y);
    fscanf(IF, "%ld %ld\n", &P[1].X, &P[1].Y);
    fscanf(IF, "%d\n", &R);
    Bus = new sBus [R];
    for (I = 0; I < R; I++) {
        fscanf(IF, "%d %d ", &Bus[I].N, &Bus[I].Prise);
        for (J = 0; J < Bus[I].N; J++)
            fscanf(IF, "%ld %ld", &Bus[I].P[J].X, &Bus[I].P[J].Y);
        fscanf(IF, "\n");
    }
    fclose(IF);
    // Az első R csúcs a buszokat, az utolsó kettő pedig a két pontot jelenti
    Graph.Set(R + 2, R + 2);
    for (I = 0; I < 2; I++) {
        for (J = 0; J < R; J++) {
            Min = inf;
            for (K = 0; K < Bus[J].N; K++) {
                D = Dist(Bus[J].P[K], Bus[J].P[(K + 1) % Bus[J].N], P[I]);
                if (D < Min)
                    Min = D;
            }
            Graph.SetItem(R + I, J, Min);
            Graph.SetItem(J, R + I, Min);
        }
    }
    Graph.SetItem(R, R + 1, Dist(P[0], P[1]));
    Graph.SetItem(R + 1, R, Dist(P[0], P[1]));
    for (I = 0; I < R; I++)
        for (J = 0; J < R; J++)
            if (I != J) {
                Min = inf;
                for (K = 0; K < Bus[I].N; K++)
                    for (L = 0; L < Bus[J].N; L++) {
                        D = Dist(Bus[I].P[K], Bus[I].P[(K + 1) % Bus[I].N],
                            Bus[J].P[L], Bus[J].P[(L + 1) % Bus[J].N]);
                        if (D < Min)
                            Min = D;
                    }
                Graph.SetItem(I, J, Min);
                Graph.SetItem(J, I, Min);
            }
    for (I = 0; I < R + 2; I++)
        Graph.SetItem(I, I, 0);

    Num = R + 2;
    Dijkstra(R);

    OF = fopen(Args[2], "wt");
    if (OF == NULL)
        return 1;
    fprintf(OF, "%ld", S[R + 1].D);
    fclose(OF);

    delete [] Bus;
    return 0;
}

// szakasz és pont távolsága
type Dist (point S0, point S1, point P) {
    // függőleges
    if (S0.X == S1.X) {
        if ((P.Y < S0.Y && S0.Y < S1.Y) || (P.Y > S0.Y && S0.Y > S1.Y))
            return Dist(S0, P);
        if ((P.Y > S1.Y && S1.Y > S0.Y) || (P.Y < S1.Y && S1.Y < S0.Y))
            return Dist(S1, P);
        return abs(P.X - S0.X);
    }
    // vízszintes
    if (S0.Y == S1.Y) {
        if ((P.X < S0.X && S0.X < S1.X) || (P.X > S0.X && S0.X > S1.X))
            return Dist(S0, P);
        if ((P.X > S1.X && S1.X > S0.X) || (P.X < S1.X && S1.X < S0.X))
            return Dist(S1, P);
        return abs(P.Y - S0.Y);
    }
    return inf;
}
// két szakasz távolsága
type Dist (point A0, point A1, point B0, point B1) {
    if (A0.X != A1.X && A0.Y != A1.Y)
        return inf;

    // Len: az A szakasz hossza
    type Len = Dist(A0, A1), Min = inf, D;
    // Step: az A szakaszon való lépkedés egysége
    // A: az A szakasz léptetésének aktuális helye a ciklusban
    point Step, A = A0;

    Step.X = (long)(A1.X - A0.X) / (long)Len;
    Step.Y = (long)(A1.Y - A0.Y) / (long)Len;

    for (int I = 0; I < Len; I++) {
        D = Dist(B0, B1, A);
        //printf("(%ld %ld) (%ld %ld) %ld\n", A.X, A.Y, Step.X, Step.Y, D);
        if (D < Min)
            Min = D;
        A.X+= Step.X;
        A.Y+= Step.Y;
    }
    return Min;
}
void Dijkstra (int Start) {
    int N = Num, A, I;
    type DD, Min;

    // Alapértékek beállítása.
    for (I = 0; I < Num; I++) {
        S[I].D = inf;
        S[I].bDone = 0;
        S[I].Prev = -1;
    }
    S[Start].D = 0L;
    while (N) {
        // Minimumkiválasztás, majd kékbe tevés
        Min = inf;
        A = -1;
        for (I = 0; I < Num; I++)
            if (!S[I].bDone && S[I].D < Min) {
                Min = S[I].D;
                A = I;
            }
        S[A].bDone = 1;
        N--;
        // Összehasonlítás a szomszédosakkal
        for (I = 0; I < Num; I++)
            if (!S[I].bDone) {
                DD = add(S[A].D, Graph.GetItem(A, I));
                if (DD < S[I].D) {
                    S[I].D = DD;
                    S[I].Prev = A;
                }
            }
    }
}
