﻿using System;
using System.IO;

namespace Szinhaz
{
    class Megrendeles
    {
        public const int MinSorszam = 1;
        public const int MaxSorszam = 1000;

        static int megrendelesekSzama = 0;

        int megrendelesSorszam;
        int kezdoSorszam;
        int vegSorszam;

        public Megrendeles(int kezdoSorszam, int vegSorszam)
        {
            if (kezdoSorszam < MinSorszam || kezdoSorszam > MaxSorszam)
                throw new ArgumentOutOfRangeException("kezdoSorszam");
            if (vegSorszam < MinSorszam || vegSorszam > MaxSorszam || vegSorszam < kezdoSorszam)
                throw new ArgumentOutOfRangeException("vegSorszam");

            this.megrendelesSorszam = ++megrendelesekSzama;
            this.kezdoSorszam = kezdoSorszam;
            this.vegSorszam = vegSorszam;
        }

        public int MegrendelesSorszam
        {
            get { return megrendelesSorszam; }
        }

        public int KezdoSorszam
        {
            get
            {
                return kezdoSorszam;
            }
            set
            {
                if (value < MinSorszam || value > MaxSorszam || value > VegSorszam)
                    throw new ArgumentOutOfRangeException("value");

                kezdoSorszam = value;
            }
        }

        public int VegSorszam
        {
            get
            {
                return vegSorszam;
            }
            set
            {
                if (value < MinSorszam || value > MaxSorszam || value < KezdoSorszam)
                    throw new ArgumentOutOfRangeException("value");

                vegSorszam = value;
            }
        }

        public int IntervallumHossz
        {
            get { return (VegSorszam - KezdoSorszam + 1); }
        }

        public bool BenneVan(int ulohely)
        {
            return (ulohely >= KezdoSorszam && ulohely <= VegSorszam);
        }

        public override string ToString()
        {
            return (KezdoSorszam + " " + VegSorszam);
        }
    }

    class MegrendelesHalmaz
    {
        const int MinKapacitas = 1;
        const int MaxKapacitas = 1000;

        int meret;
        Megrendeles[] megrendelesek;

        public MegrendelesHalmaz(int kapacitas)
        {
            if (kapacitas < MinKapacitas || kapacitas > MaxKapacitas)
                throw new ArgumentOutOfRangeException("kapacitas");

            meret = 0;
            megrendelesek = new Megrendeles[kapacitas];
        }

        public int Meret
        {
            get { return meret; }
        }

        public Megrendeles this[int index]
        {
            get
            {
                if (index < 0 || index >= meret)
                    throw new ArgumentOutOfRangeException("index", "Érvénytelen index.");

                return megrendelesek[index];
            }
        }

        public void Hozzaad(Megrendeles megrendeles)
        {
            if (meret >= megrendelesek.Length)
                throw new InvalidOperationException("Nincs több hely az új elemnek.");
            
            megrendelesek[meret++] = megrendeles;
        }

        public void Torol(int index)
        {
            if (index < 0 || index >= meret)
                throw new ArgumentOutOfRangeException("index", "Érvénytelen index.");

            megrendelesek[index] = megrendelesek[--meret];
        }
    }

    public class Szervezo
    {
        int ulohelyekSzama;
        bool megrendelesekLezarva;
        MegrendelesHalmaz megrendelesek;

        public Szervezo(int ulohelekSzama, int megrendelesekSzama)
        {
            if (ulohelekSzama < Megrendeles.MinSorszam || ulohelekSzama > Megrendeles.MaxSorszam)
                throw new ArgumentOutOfRangeException("ulohelekSzama");

            this.ulohelyekSzama = ulohelekSzama;
            this.megrendelesekLezarva = false;
            this.megrendelesek = new MegrendelesHalmaz(megrendelesekSzama);
        }

        public int UlohelyekSzama
        {
            get { return ulohelyekSzama; }
        }

        public void MegrendelestFogad(int kezdoSorszam, int vegSorszam)
        {
            if (kezdoSorszam > UlohelyekSzama)
                throw new ArgumentOutOfRangeException("kezdoSorszam");
            if (vegSorszam > UlohelyekSzama)
                throw new ArgumentOutOfRangeException("vegSorszam");

            megrendelesek.Hozzaad(new Megrendeles(kezdoSorszam, vegSorszam));
        }

        private void MegrendeleseketLezar()
        {
            if (megrendelesekLezarva)
                return;

            for (int ulohely = 1; ulohely <= UlohelyekSzama; ulohely++) {
                int minHossz = int.MaxValue;
                int minIndex = 0;
                bool talalt = false;

                for (int i = 0; i < megrendelesek.Meret; i++) {
                    if (megrendelesek[i].BenneVan(ulohely)) {
                        if (minHossz > megrendelesek[i].IntervallumHossz) {
                            minHossz = megrendelesek[i].IntervallumHossz;
                            minIndex = i;
                            talalt = true;
                        }
                    }
                }

                if (talalt) {
                    for (int i = 0; i < megrendelesek.Meret; i++) {
                        if (megrendelesek[i].BenneVan(ulohely)) {
                            if (i == minIndex)
                                megrendelesek[i].VegSorszam = megrendelesek[i].KezdoSorszam;
                            else if (megrendelesek[i].IntervallumHossz >= 2)
                                megrendelesek[i].KezdoSorszam++;
                            else
                                megrendelesek.Torol(i--);
                        }
                    }
                }
            }

            megrendelesekLezarva = true;
        }

        public void MegrendeleseketKiir(TextWriter writer)
        {
            if (writer == null)
                throw new ArgumentNullException("writer");

            MegrendeleseketLezar();

            writer.WriteLine(megrendelesek.Meret);
            for (int i = 0; i < megrendelesek.Meret; i++) {
                writer.WriteLine("{0} {1}", megrendelesek[i].MegrendelesSorszam,
                    megrendelesek[i].KezdoSorszam);
            }
        }
    }

    public class Program
    {
        private static Szervezo Beolvas(string fileNev)
        {
            using (StreamReader reader = new StreamReader(fileNev))
            {
                string[] elsoSor = reader.ReadLine().Split();
                int m = int.Parse(elsoSor[0]);
                int n = int.Parse(elsoSor[1]);
                Szervezo szervezo = new Szervezo(m, n);

                for (int i = 0; i < n; i++) {
                    string[] sor = reader.ReadLine().Split();
                    int a = int.Parse(sor[0]);
                    int b = int.Parse(sor[1]);

                    szervezo.MegrendelestFogad(a, b);
                }
  
                return szervezo;
            }
        }

        public static void Main(string[] args)
        {
            if (args.Length > 0) {
                Szervezo szervezo = Beolvas(args[0]);
                szervezo.MegrendeleseketKiir(Console.Out);
            }
            else {
                Console.WriteLine("Használat: szinhaz <inputfile>");
            }            
        }
    }
}
