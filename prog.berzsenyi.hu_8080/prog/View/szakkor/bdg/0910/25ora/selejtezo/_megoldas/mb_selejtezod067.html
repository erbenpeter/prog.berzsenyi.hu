    beolvas("C-large-practice.in", "C-large-practice.out");

}

long long solve()
{
    bool cont=true;
    long long pmax, nmax;
    pmax=-1;
    while(cont)
    {
        nmax=0;
        for(int i=0; i<p; i++)
            nmax+=s[i];
        nmax=nmax/c;
        if(nmax==pmax)
        {
            cont=false;
            break;
        }
        
        for(int i=0; i<p; i++)
            if(s[i]>nmax)
                s[i]=nmax;
            
        pmax=nmax;
    }
    return pmax;
}
