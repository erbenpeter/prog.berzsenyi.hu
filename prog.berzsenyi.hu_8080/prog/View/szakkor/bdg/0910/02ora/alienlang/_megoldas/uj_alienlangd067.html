    Informatika: algoritmus szakkor
    Feladat: Idegen nyelv
    Uray M. Janos, 2009. szeptember 22.
    Forras: http://prog.berzsenyi.hu:8080/prog/View/szakkor/bdg/0910/02ora/alienlang

#include <stdio.h>

#define MaxL 15
#define MaxD 5000
#define MaxN 500

// A nyelv szavai
char Dict[MaxD][MaxL + 1];
int D;

int Count (const char *);

int main (int AN, char * Args []) {
    // Test[]: egy teszteset stringkent
    char Test[30 * MaxL];
    int L, N, I;
    FILE * IF, * OF;
    
    // Parancssori argumentumok ellenorzese, allomanyok megnyitasa
    if (AN < 2)
        return 1;
    IF = fopen(Args[1], "r");
    if (IF == NULL)
        return 1;
    OF = fopen(Args[2], "w");
    if (OF == NULL) {
        fclose(IF);
        return 1;
    }
    
    // Adatok beolvasasa, feldolgozasa es mentese
    fscanf(IF, "%d %d %d\n", &L, &D, &N);
    for (I = 0; I < D; I++)
        fscanf(IF, "%s\n", Dict[I]);
    for (I = 0; I < N; I++) {
        fscanf(IF, "%s\n", Test);
        fprintf(OF, "Case #%d: %d\n", I + 1, Count(Test));
    }
    
    // Allomanyok bezarasa
    fclose(IF);
    fclose(OF);
    
    return 0;
}

/* Teszteset feldolgozasa
    Test: a teszteset stringkent
    Dict: szavak tombje
    D: a szotari szavak szama
*/
int Count (const char * Test) {
    // Q: a fuggveny visszateresi erteke: a megfelelo szavak szama
    int I, J, Q = 0;
    // A teszteset olvasasakor eppen zarojelben vagyunk-e
    bool bBracket = false;
    // A teszteset konvertalt valtozata: az adott sorszamu helyen az adott betu lehetseges-e
    bool TestB[MaxL][26];
    
    // TestB kiuritese
    for (I = 0; I < MaxL; I++)
        for (J = 0; J < 26; J++)
            TestB[I][J] = false;
    
    // Teszteset konvertalasa TestB[][]-be.
    // J: a token sorszama
    J = 0;
    for (I = 0; Test[I]; I++) {
        switch (Test[I]) {
            case '(':
                bBracket = true;
                break;
            case ')':
                bBracket = false;
                J++;
                break;
            default:
                if (Test[I] >= 'a' && Test[I] <= 'z')
                    TestB[J][Test[I] - 'a'] = true;
                if (!bBracket)
                    J++;
        }
    }
    
    /* Szotar minden szavanak osszehasonlitasa a teszttel. Ha egy szonak egy betuje is van,
    amely nem illik a tesztesetbe, akkor a belso ciklus megszakad. Ha a ciklus veget ert,
    akkor a jo szavak szamat noveljuk. */
    for (I = 0; I < D; I++) {
        for (J = 0; Dict[I][J]; J++)
            if (!TestB[J][Dict[I][J] - 'a'])
                break;
        if (!Dict[I][J])
            Q++;
    }
    
    return Q;
}
