function volt(s:TState):boolean;
var i,j:longint;
begin
  {
  i:=voltdb;
  while (i>0) and gt(voltt[i],s) do dec(i);
  
    
  if (i=0) or not(eq(voltt[i],s)) then
    begin
      //beszúrni i+1. helyre
      if voltdb=voltsize then
        begin
          voltsize := voltsize + 100;
          setlength(voltt,voltsize+1);
        end;
      
      for j:=voltdb downto i+1 do voltt[j+1]:=voltt[j];
      voltt[i+1]:=s;
      inc(voltdb);
      volt:=false;
      writeln('False:',s.red,',',s.other);
    end
  else
    begin
      volt:=true;
    end;
  }
  i:=voltdb;
  while (i>0) and not(eq(voltt[i],s)) do dec(i);
  if i=0 then
    begin
      volt:=false;
      if voltdb=voltsize then
        begin
          voltsize := voltsize + 100;
          setlength(voltt,voltsize+1);
        end;
      inc(voltdb);
      voltt[voltdb]:=s;      
    end
  else
    begin
      volt:=true;
    end;    
end;

end;
