Minimum kiválasztásos rendezés

Kattintásra indul a rendezés.
void sort(int T[]){
    for (int i = 0; i<T.length - 1; i++) {
        int min = i;
        for (int j = i+1; j<T.length; j++) {
            if (T[j] < T[min]) { min = j;}
        }
        if( i != min) {
            int tmp = T[i];
            T[i] = T[min];
            T[min] = tmp;
        }
    }
}
