using System;
using System.IO;

namespace Darabolas
{
    class Program
    {
        
        int[,,] cache = new int[21,21,21];

        private int opt( int x, int y, int db ) {
            
            int v1 = 0;
            int v2 = 0;
            int res = Int32.MaxValue;
            
            if( cache[x, y, db] != 0 ) { 
                return cache[x, y, db];
            }
            
            if( x*y < db ) return Int32.MaxValue;
            if( db == 1 ) return x*y;
                    
                // d tartalmazza, hogy mennyit vágunk még az első darabból, és mennyit a másodikból.
                
                for( int a=1; a<x; a++ ) {
                    for( int d=1; d<db; d++ ) {
                        v1 = opt( a, y, db-d );
                        v2 = opt( x-a, y, d );
                        if( v1>v2 ) {
                            if( v1<res ) res = v1;
                        } else {
                            if( v2<res ) res = v2;
                        }
                    }
                }
                
                for( int b=1; b<y; b++ ) {
                    for( int d=1; d<db; d++ ) {
                        v1 = opt( x, b, db-d );
                        v2 = opt( x, y-b, d );
                        if( v1>v2 ) {
                            if( v1<res ) res = v1;
                        } else {
                            if( v2<res ) res = v2;
                        }
                    }
                }
                
            cache[x, y, db] = res;
            return res;
                
        }
        
        private void LetsDO() {
            StreamReader sr = new StreamReader("darabol.be");
            
            string line;
            // Read and display lines from the file until the end of 
            // the file is reached.
            string[] input;
            TextWriter tw = new StreamWriter("darabol.ki");

            while ((line = sr.ReadLine()) != null) 
            {
                input = line.Split(' ');
                // x, y, db
                for(int a = 0; a<21; a++) {
                    for(int b=0; b<21; b++) {
                        for(int c=0; c<21; c++) {
                            cache[a, b, c] = 0;
                        }
                    }
                }
                
                if( Int32.Parse(input[0]) == 0 && Int32.Parse(input[1]) == 0 && Int32.Parse(input[2]) == 0 ) {
                    break;
                }
                
                tw.WriteLine( opt( Int32.Parse(input[0]), Int32.Parse(input[1]), Int32.Parse(input[2]) ) );
            }   

            tw.Close();
            sr.Close();

        }
        
        public static void Main(string[] args)
        {
            
            Program prg = new Program();
            prg.LetsDO();           
            
        }
    }
}
