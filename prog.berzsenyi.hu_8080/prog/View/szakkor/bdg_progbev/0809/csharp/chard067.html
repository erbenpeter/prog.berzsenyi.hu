char

A karaktereket tároló változók típusa. Egy karakter pontosan egy darab betűt vagy számot vagy egyéb írásjelet tartalmaz. A karakter-értékeket mindig szimpla-idézőjelek közé kell tenni:
char ch;
ch = 'a';

Példák

using System;

class Program
{
    public static void Main(string[] args)
    {
        Console.Write("Üss le egy billentyűt, aztán üss entert: ");
        char ch = char.Parse( Console.ReadLine() );
        
        Console.WriteLine("Ez volt az: "+ch);
        
        if (char.IsLetter(ch)) {
            Console.WriteLine("Ez egy betű.");
            if (char.IsUpper(ch)) {
                Console.WriteLine("Méghozzá nagybetű.");
            } else if (char.IsLower(ch)) {
                Console.WriteLine("Méghozzá kisbetű.");
            }
        }
        Console.ReadKey(true);
    }
}
