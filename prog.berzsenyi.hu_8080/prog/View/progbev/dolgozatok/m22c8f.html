2. feladat megoldása

Mikó Péter kódja alapján:
program f2;
var a,c,d,i:longint;
begin
readln (a);
for i := 1 to 1000000 do
    begin
    d:=0;
    for c:=1 to i do {szigma(i) kiszámítása}
        begin
        if i mod c = 0 then d:=d+c;
        end;
    if d=a then writeln (i);    
    end;
end.
