3. feladat megoldása

program f3;
{n -> n/2-1 (ha n ps) v. n - 17 (ha n ptln)}
var N : longint;
begin
    readln(N);
    while N >= 10 do
    begin
        if (N mod 2) = 0
            then N := N div 2 -1
            else N := N - 17;           
    end;
    writeln(N);
end.
