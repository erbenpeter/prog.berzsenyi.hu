package hu.krivan.szakkor.dijkstra;

/**
 *
 * @author Bálint Kriván
 */
public abstract class Node implements Comparable<Node> {

    protected int distance = Integer.MAX_VALUE;
    protected Node daddy;

    public final int getDistance() {
        return distance;
    }

    public final void setDistance(int i) {
        distance = i;
    }

    public final Node getDaddy() {
        return daddy;
    }

    public final void setDaddy(Node daddy) {
        this.daddy = daddy;
    }

    public final int compareTo(Node o) {
        if (this.getDistance() < o.getDistance()) {
            return -1;
        } else if (this.getDistance() == o.getDistance()) {
            return 0;
        } else {
            return 1;
        }
    }

    /**
     * A gráf csúcs távolsága egy adott gráf csúcstól, magyarul a két csúcs közötti él súlya
     * @param o gráf csúcs
     * @return két csúcs közötti távolság
     */
    public final int getDistanceFrom(Node o) {
        if (o instanceof PointNode) {
            return getDistanceFrom((PointNode) o);
        } else /*if (o instanceof RouteNode)*/ {
            return getDistanceFrom((RouteNode) o);
        }
    }

    public final void traceBackToStart() {
        Node n = this;
        System.out.print(n);
        while (n.getDaddy() != null) {
            n = n.getDaddy();
            System.out.print(" -> " + n);
        }
        System.out.println();
    }

    /**
     * A gráf csúcs távolsága egy adott <b>pont</b> típusú csúcstól
     * @param o pont típusú csúcs
     * @return távolság
     */
    public abstract int getDistanceFrom(PointNode o);

    /**
     * A gráf csúcs távolsága egy adott <b>(busz)útvonal</b> típusú csúcstól
     * @param o útvonal típusú csúcs
     * @return távolság
     */
    public abstract int getDistanceFrom(RouteNode o);

    @Override
    public abstract String toString();
}
