using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

namespace munka
{
    class Job : IComparable<Job>
    {
        public int id, time;

        public Job(int _time, int _id)
        {
            time = _time;
            id = _id;
        }

        public int CompareTo(Job other)
        {
            return -time.CompareTo( other.time ); // hosszabb kell nekünk előre!
        }
    }

    class munka
    {
        int n;
        List<Job> jobs;
        List<int> g1, g2;
        int t1 = 0, t2 = 0;

        public munka()
        {
            StreamReader ins = new StreamReader("munka.be");
            n = int.Parse(ins.ReadLine());
            string[] line = ins.ReadLine().Split(' ');
            jobs = new List<Job>(n);
            g1 = new List<int>(n);
            g2 = new List<int>(n);
            for (int i = 0; i < n; i++)
            {
                jobs.Add( new Job( int.Parse(line[i]), i+1 ) );
            }
            jobs.Sort();

            selecting();
            printOut();
        }

        private void printOut()
        {
            StreamWriter outs = new StreamWriter("munka.ki");
            outs.WriteLine( Math.Max( t1, t2 ) );
            for (int i = 0; i < g1.Count; i++) outs.Write(g1[i] + " ");
            outs.WriteLine();
            for (int i = 0; i < g2.Count; i++) outs.Write(g2[i] + " ");
            outs.WriteLine();
            outs.Close();
        }

        private void selecting()
        {
            for (int i = 0; i < n; i++)
            {
                if (t1 >= t2)
                {
                    t2 += jobs[i].time;
                    g2.Add(jobs[i].id);
                }
                else
                {
                    t1 += jobs[i].time;
                    g1.Add(jobs[i].id);
                }
            }
        }         

        static void Main(string[] args)
        {
            new munka();
        }
    }
}
