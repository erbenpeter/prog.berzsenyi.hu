void sort(int T[]){
    for (int i = 0; i<T.length - 1; i++ ) {
        for (int j = i + 1; j<T.length; j++) {
            if (T[i] > T[j]) {
                int tmp = T[i]; T[i] = T[j]; T[j] = tmp;
            }
        }
    }
}
