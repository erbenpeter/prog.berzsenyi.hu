package hu.krivan.szakkor.dijkstra;

import java.io.BufferedReader;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;

/**
 *
 * @author Bálint Kriván
 */
public class Routing {

    /**
     * Minimálisan megtehető lépésszám.
     */
    private int d;
    /**
     * Az A csúcs
     */
    private PointNode nodeA;
    /**
     * A B csúcs
     */
    private PointNode nodeB;
    /**
     * Busz útvonalak száma
     */
    private int r;
    /**
     * Busz útvonalak
     */
    private ArrayList<RouteNode> routes;

    Routing() throws FileNotFoundException, IOException {
        BufferedReader in = new BufferedReader(new FileReader("input.txt"));

        d = Integer.valueOf(in.readLine());
        nodeA = new PointNode(in.readLine());
        nodeB = new PointNode(in.readLine());
        r = Integer.valueOf(in.readLine());
        routes = new ArrayList<RouteNode>();
        for (int i = 0; i < r; i++) {
            routes.add(new RouteNode(in.readLine()));
        }
    }

    private void generateGraph() {
        Graph g = new Graph();
        // fontos a sorrend! kezdő, végző és a többi.
        g.addNode(nodeA); // kezdő
        g.addNode(nodeB); // végző
        g.addNodes(routes);
        g.findShortestWay();
    }

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) throws FileNotFoundException, IOException {
        Routing r = new Routing();
        r.generateGraph();
    }
}
