PROGRAM fibo2;
USES SysUtils;
VAR
    a, b, c, i: INTEGER;
BEGIN
    a:= 1;
    b:= 1;
    FOR i:= 3 TO 44 DO
    BEGIN
        c:= a+b;
        WriteLn('fib(',i,')= ', c);
        a:= b; b:= c;
    END;
    ReadLn;
END.
