/**
 * A megoldó osztály, ő végzi a koordináló feladatokat.
 * 
 * @author Kriván Bálint
 */
public class Solution {
    
    /**
     * A singleton osztályunk egyetlen szál példányát őrzi.
     */
    private static Solution instance = null;
    
    /**
     * A már vizsgált állapotokat tároljuk benne (tömörítve, nem State objektumként)
     */
    public CheckedStates checkedStates = new CheckedStates();
    
    /**
     * Lekérjük a példányunkat.
     * 
     * @return Solution
     */
    public static Solution getInstance()
    {
        if( instance == null ) {
            instance = new Solution();            
        }
        return instance;
    }
    
    /**
     * Probléma megoldó metódusunk.     
     */
    public void solve()
    {        
        // Kezdő állapot:
        State state = new State();
        // Sor inicializálása
        Queue<State> queue = new Queue<State>();
        queue.add(state);               

        // amíg nem a végső állapotban vagyunk, addig keressük a végső állapotot.
        while( !state.isFinal() ) {
            // lekérjük a sor elejéről az állapotot
            state = queue.pop();
            // majd meghatározzuk ennek az állapotból elérhető következő állapotokat
            queue.addAll( state.getNextStates() );
        }
        
        /*
         * Kész vagyunk! Elértük a végső állapotot, már csak szeretnénk tudni, hogy pontosan
         * hogy is értük el ;)
         */                
        
        // Legutolsó állapot tömörítve
        long node = state.zip();
        // Legelső állapot tömörítve
        long first = new State().zip();        
        int steps = 1;
        while( true ) {
            // lekérjük a node ősét, amit a node-ban tárolunk.
            node = Solution.getInstance().checkedStates.relations.get( node );
            // ha ő a kiinduló állapot, akkor kész vagyunk, és nem számulunk tovább
            if( node == first ) {
                break;
            }
            steps++;
        }
        
        // kiiratjuk a mágikus lépésszámot
        System.out.println( steps );
                
    }
    
}

/**
 * A már vizsgált állapotok tárolásához szükséges objektum.
 * 
 * @author Kriván Bálint
 */
public class CheckedStates extends ArrayList<Long> {

    /**
     * Ez tárolja, hogy melyik állapotot, melyik előző állapotból értük el.
     */
    public HashMap<Long, Long> relations = new HashMap<Long, Long>(); 
    
    /**
     * Speckó add metódus: nem akárhova szúrja be az elemet, hanem pontosan a helyére
     * azaz, rendezetten. Ha már létezik elem, akkor nem tudtuk beszúrni.
     * 
     * @param   zstate  az állapot
     * @param   fater   az állapot, amelyből elértük a <I>zstate</I> állapotot
     * @return      beszúrás sikeressége (true/false). (sikertelen, ha már létezik ilyen állapot)
     */
    public boolean add( Long zstate, Long fater )
    {        
        int pos = getOptPos( zstate );
        if( pos != -1 ) {
            relations.put(zstate, fater);
            super.add( pos, zstate );
            return true;
        } else return false;
    }
    
    /**
     * Lekérjük az adott állapot optimális helyét a listában
     * 
     * @param   zipped  az állapot, aminek a helyét keressük (tömörítve)
     * @return          (-1)-el tér vissza ha már létezik az elem, egyébként pedig
     *                  az optimális pozícióval a listában (optimális = rendezett,
     *                  legyen a lista)
     */
    private int getOptPos( long zipped ) {
        /*
         * Most jön egy kis finom bináris keresés, hiszen tudnunk, hogy hova tegyük
         * az adott állapotot a listában.
         */
        
        if( this.size() == 0 ) return 0; // értelemszerűen.
        
        int low = 0;
        int high = this.size() - 1;
        int mid = 0;
        
        while( low <= high )
        {
            mid = (low + high) / 2; // bökjünk a vizsgált tartomány közepébe.
            if( this.get(mid) < zipped ) low = mid+1; // tőle jobbra lesz megoldás
            else if( this.get(mid) > zipped ) high = mid-1; // tőle balra lesz megoldás
            else return -1; // Ilyen jelen esetben nem lehetséges, hiszen nem szúrhatunk be már létező cuccost!
        }
        
        return low;
    }
}

/**
 * Ez az objektum reprezentálja a játék egy állapotát.
 * 
 * @author Kriván Bálint
 */
public class State {
    
    /**
     * Tárolja az elemek pozícióját
     */
    public int[] objects = {
                             0,  3, 12, 15, // fekete cuccosok
                             9,             // szürke
                             13, 14, 17, 18, // piros cuccosok
                             1,             // SÁRGA!!!
                             8, 11          // a két üres spot
                           };
    
    /**
     * Ez határozza meg, hogy melyik elem melyik csoportba tartozik.
     * Tehát a fekete elemek, azok 1 csoport, a szürke is egy, a pirosok is egy, stb...
     */
    public int[] objtypes = {
                                1, 1, 1, 1,
                                2,
                                3, 3, 3, 3,
                                4,
                                0, 0
                            };
    
    /**
     * Becsomagoljuk az állapotunkat egy 60 bites longba.
     * @return tömörített állapot
     */
    public long zip()
    {
        long res = 0;        
        
        /*
         * Jelen pillanatban nem látok jobb megoldást:
         * dobjuk ki a cuccost egy 2d-s tömbbe, de úgy, hogy azonos típusú
         * objektumokat azonos számmal jelöljük, majd ezeken menjünk végig
         * és hozzuk össze a 20*3 = 60bites longot.
         */
        
        long[][] tmp = new long[4][5];
        
        int obj;     
        int a, b;
        for( int i=0; i<objects.length; i++ ) {
            obj = objects[i];
            a = obj % 4;
            b = obj / 4;
            for( int x = a; x<a+Board.widths[i]; x++ ) {
                for( int y = b; y<b+Board.heights[i]; y++ ) {
                    tmp[x][y] = objtypes[i];
                }
            }
        }       
        
        /*
         * van egy 4*5-ös táblánk, amiben el van tárolva, hogy melyik pozíción,
         * milyen típusú/csoportú elemünk van. Ezt a táblát sorfolytonosan ábrázoljuk
         * egy longban.
         */
        
        for( int x=0; x<4; x++ ) {
            for( int y=0; y<5; y++ ) {
                res |= tmp[x][y] << (long)((x+y*4)*3);
            }
        }       
                
        return res;
    }
    
    /**
     * Kicsomagoljuk az állapotot egy Board-ba
     * 
     * @return  kicsomagolt Board
     * @see     Board
     */
    public Board extract()
    {
        return new Board( this.objects );
    }

    /**
     * Értelemszerűen meghatározza, hogy az adott állapot jó-e nekünk, ahhoz hogy befejezzük.
     * 
     * @return true, ha az adott állapotunk végeredményhez vezet, false, ha nem.
     */
    public boolean isFinal() {
        /*
         * praktikusan mikor mondható végsőnek egy állapot?
         * Ha a sárga elem a kivezető helyen van, azaz: a 13. pozíció.
         */
        
        return ( objects[9] == 13 );
    }
    
    /**
     * Meghatározza az adott állapotból következő állapotokat, figyelembe véve,
     * hogy volt-e már az adott állapot checkolva. Ha igen, akkor azt már köszönjük,
     * de nem kérjük.
     * 
     * @return következő állapotok
     */
    public State[] getNextStates() {
        State[] states = null;
        
        // kicsomagoljuk az állapotot egy Board-ba, majd megkeressük a táblából elérhető köverkező táblákat
        Board[] boards = this.extract().getNextBoards();
        
        states = new State[ boards.length ];

        for( int i=0; i<boards.length; i++ )
        {
            // mind egyes táblát visszakonvertálunk állapotba.
            states[i] = boards[i].toState();
        }
        
        return states;
    }
    
}

import java.util.Vector;

/**
 * Megvalósítjuk a sort, ami jelen esetben az állapotok tárolásához kell.
 * 
 * @author Kriván Bálint
 */
public class Queue<E> extends Vector<E> {

    /**
     * Feltúrbozott beszúrás, hiszen egy egész tömböt beszúrunk.
     * 
     * @param elements  beszúrandó elemek tömbje
     * @return true
     */
    public boolean addAll(E[] elements) {
        boolean res = true;
        
        for( int i=0; i<elements.length; i++ )
        {
            res = (res && this.add( elements[i] ));
        }
        
        return res;
    }
    
    /**
     * Kivesszük az első elemet a sorból (aki leghosszabb ideje van bent)
     * 
     * @return kivett elem
     */
    public E pop() {
        return this.remove( 0 );        
    }

    /**
     * Csak meglessük az első elemet, de nem vesszük ki
     * 
     * @return kivehető elem
     */
    public E peek() {
        return this.get( 0 );        
    }

}

/**
 * Ez reprezentál egy táblát, tehát egy állapotot, de táblában kicsomagolva.
 * 
 * @author Kriván Bálint
 */
public class Board {

    /**
     * Ez a 2D-s tömb tárolja primitív formában a táblát.
     * Főleg a mozgatás meghatározásához kell.
     */
    private int[][] holder = new int[4][5];
    
    /**
     * Az állapothoz tartozó elemek pozíciója
     * @see State#objects
     */
    private int[] objects;
    
    private final int UP = 2;
    private final int LEFT = 1;
    private final int RIGHT = 3;
    private final int DOWN = 4;
    
    /**
     * A táblán található elemek szélességeit tartalmazza, hiszen tudnunk kell, 
     * hogy melyik elem milyen széles, ahhoz, hogy megfelelően lehessen mozgatni.
     */
    public static final int[] widths = {
                           1, 1, 1, 1,
                           2,
                           1, 1, 1, 1,
                           2,
                           1, 1
                          };
    
    /**
     * A táblán található elemek magasságait tartalmazza, hiszen tudnunk kell, 
     * hogy melyik elem milyen magas, ahhoz, hogy megfelelően lehessen mozgatni.
     */
    public static final int[] heights = {
                            2, 2, 2, 2,
                            1,
                            1, 1, 1, 1,
                            2,
                            1, 1
                           };
        
    /**
     * Felépítjük a Board-ot az elemek helyzetéből
     * 
     * @param objects  A tábla generálásához szükséges elemek helyzete
     * @see   State#objects
     */
    public Board( int[] objects )
    {
        // Ide jön a kitöltés...
        int i;
        int x;
        int y;
        // feketék        
        for( i=0; i<4; i++ )
        {
            x = getX(objects[i]);
            y = getY(objects[i]);
            holder[ x ][ y ] = i;
            holder[ x ][ y+1 ] = i;
        }
        // szürke
        x = getX(objects[4]);
        y = getY(objects[4]);
        holder[ x ][ y ] = 4;
        holder[ x+1 ][ y ] = 4;
        // piros
        for( i=5; i<9; i++ )
        {
            x = getX(objects[i]);
            y = getY(objects[i]);
            holder[ x ][ y ] = i;
        }
        // sárga
        x = getX(objects[9]);
        y = getY(objects[9]);
        holder[ x ][ y ] = 9;
        holder[ x+1 ][ y ] = 9;
        holder[ x ][ y+1 ] = 9;
        holder[ x+1 ][ y+1 ] = 9;
        
        for( i=10; i<12; i++ )
        {
            x = getX(objects[i]);
            y = getY(objects[i]);
            holder[ x ][ y ] = -1;
        }
        
        // elmentjük az elemek helyzetét, hogy később a táblánk is tudja, hogy mi a szitu.
        this.objects = objects.clone();
    }

    /**
     * Visszaadja az adott táblából elérhető (tehát 1 mozgatásból megkapható) táblákat.
     * 
     * @return a táblából elérhető következő táblák
     */
    public Board[] getNextBoards() {
        
        int x;
        int y;
        int objid;

        ArrayList<Integer> movings = new ArrayList<Integer>();
        
        // lekérjük a két üres pozíciót és megnézzük a szomszédokat
        for( int i=10; i<12; i++ )
        {
            x = getX( objects[i] );
            y = getY( objects[i] );
            
            // szomszéd vizsgálat, majd hozzáadjuk a movings listához, ha lehet az adott cuccost mozgatni
            if (x > 0) {
                objid = holder[x - 1][y];                
                addAsMovable( objid, RIGHT, movings );
            }
            if (x < 3) {
                objid = holder[x + 1][y];
                addAsMovable( objid, LEFT, movings );
            }
            if (y > 0) {
                objid = holder[x][y - 1];
                addAsMovable( objid, DOWN, movings );
            }
            if (y < 4) {
                objid = holder[x][y + 1];
                addAsMovable( objid, UP, movings );
            }
        }
        
        int move;
        ArrayList<Board> boards = new ArrayList<Board>();
        Board tmp_board = null;
       
        /*
         * Most vesszük sorra a mozgásokat (minden mozgás egyedi lesz, nincs 2 azonos)
         * És legeneráljuk a táblákat (biztos, hogy még nem vizsgált állapot tábláját fogjuk legenerálni
         */
        
        long prev_zipped_state = this.toState().zip();
        
        for( int i=0; i<movings.size(); i++ ){
            move = movings.get(i);
            tmp_board = generateBoardFromMove( move );
            
            if( Solution.getInstance().checkedStates.add( tmp_board.toState().zip(), prev_zipped_state ) )
            {
                // csak akkor adjuk a még vizsgálandó táblákhoz az adott táblát, ha az még nem volt checkolva!
                boards.add( tmp_board );                
            }
        }

        return boards.toArray( new Board[ boards.size() ] );                
    }

    /**
     * Ha az adott elemet lehet mozgatni az adott irányba, akkor hozzáadja a
     * mozgatások listájához az adott mozgatást.
     * 
     * @param objid      az elem IDja, amit mozgatni szeretnénk
     * @param direction  az irány amerre az elemet majd mozgatjuk
     * @param movings    az eddigi mozgatások listája
     */
    private void addAsMovable(int objid, int direction, ArrayList<Integer> movings) {        
        int move;
        
        if (objid != -1) { // ha nem üres helyről van szó, hiszen őt nem mozgatjuk
            if (isMovable(objid, direction)) { // ha mozgatható az adott elem az adott irányban
                move = objid | (direction << 4);
                if (!movings.contains(move)) { // ha még nem volt ilyen mozgatás
                    // akkor hozzácsapjuk a mozgatások listájához ezt a mozgatást is.
                    movings.add(move);
                }
            }
        }        
    }

    /**
     * A 2D-s indexelést átkonvertáljuk sorfolytonos indexbe.

     * @param   x   (x) index (0-4)
     * @param   y   (y) index (0-5)
     * @return      sorfolytonos index (0-19)
     */
    private int getPos(int x, int y) {
        return y*4 + x;
    }
    
    /**
     * A sorfolytonos indexelést átkonvertálja 2D-s indexelésbe, ahol is az <I>x</I> indexre vagyunk kiváncsiak
     * 
     * @param   pos     az adott mező
     * @return          a Board adott mezőjének x indexe

    private int getX( int pos )

    /**

     * @return          a Board adott mezőjének x indexe

    private int getY( int pos )

    /**
     * Mozgatható az adott test az adott irányba?
     * 
     * @param   obj     az elem IDja, amit mozgatnánk
     * @param   irany   az irany amerre mozgatnánk
     * @return  true, ha mozgatható az adott obj, irányban
     */
    private boolean isMovable(int obj, int irany) {
                
        // meghatározzuk az elem balfelső csúcsának koordintáit a táblában.
        int x = getX( objects[obj] );
        int y = getY( objects[obj] );
        
        /*
         * Megvizsgáljuk, hogy a megfelelő irányban történő elmozgatás lehetésges-e:
         *  - Ha az elemnek a megfelelő dimenziója csak 1 hosszú, akkor természetesen betolható oda az elem.
         *  - Ha az elemnek több helyre (2) van szüksége a sikeres mozgatáshoz, akkor ellenőrizzük, hogy
         *    a kellő hely meglétét.
         */
        switch( irany ) {
                case LEFT:
                    if( holder[x-1][y] == -1 ) {
                        if( heights[obj] == 1 ) return true; else
                        if( heights[obj] == 2 ) {
                            if( holder[x-1][y+1] == -1 ) return true;
                        }
                    }
                    break;
                case UP:
                    if( holder[x][y-1] == -1 ) {
                        if( widths[obj] == 1 ) return true; else
                        if( widths[obj] == 2 ) {
                            if( holder[x+1][y-1] == -1 ) return true;
                        }
                    }
                    break;
                case RIGHT:
                    if( holder[x+widths[obj]][y] == -1 ) {
                        if( heights[obj] == 1 ) return true; else
                        if( heights[obj] == 2 ) {
                            if( holder[x+widths[obj]][y+1] == -1 ) return true;
                        }
                    }
                    break;
                case DOWN:
                    if( holder[x][y+heights[obj]] == -1 ) {
                        if( widths[obj] == 1 ) return true; else
                        if( widths[obj] == 2 ) {
                            if( holder[x+1][y+heights[obj]] == -1 ) return true;
                        }
                    }
                    break;                
        }        
        
        // a mozgatás nem volt lehetséges:
        return false;
    }
    
    /**
     * Az adott mozgatásból legenerálja a jelenlegi táblából a következő táblát
     * 
     * @param move  mozgatás
     * @return      a mozgatással megkapott tábla
     */
    private Board generateBoardFromMove( int move  )
    {
        // kitaláljuk, hogy az adott mozgatás, milyen objektumot, milyen irányban mozgat
        int direction = (move >> 4);
        int obj = move & (~(direction << 4));            
        
        int[] newobjects = this.objects.clone();
        
        switch( direction ) {
            case UP:
                newobjects[ obj ] -= 4;
                break;
            case DOWN:
                newobjects[ obj ] += 4;
                break;
            case RIGHT:
                newobjects[ obj ]++;
                break;
            case LEFT:
                newobjects[ obj ]--;
                break;
        }
        
        // üres spotokat állítsuk vissza!!!
        // TODO - ezen esetleg improveolni, hogy ne kelljen az egész táblát átcheckolni...
        setEmpties( newobjects );                
        
        Board res = new Board( newobjects );        
        return res;
    }

    private void setEmpties(int[] objects) {
        int[][] array = new int[4][5];
        
        int i;
        int x;
        int y;
        // feketék        
        for( i=0; i<4; i++ )
        {
            x = getX(objects[i]);
            y = getY(objects[i]);
            array[ x ][ y ] = 1;
            array[ x ][ y+1 ] = 1;
        }
        // szürke
        x = getX(objects[4]);
        y = getY(objects[4]);
        array[ x ][ y ] = 1;
        array[ x+1 ][ y ] = 1;
        // piros
        for( i=5; i<9; i++ )
        {
            x = getX(objects[i]);
            y = getY(objects[i]);
            array[ x ][ y ] = 1;
        }
        // sárga
        x = getX(objects[9]);
        y = getY(objects[9]);
        array[ x ][ y ] = 1;
        array[ x+1 ][ y ] = 1;
        array[ x ][ y+1 ] = 1;
        array[ x+1 ][ y+1 ] = 1;
        
        //most az array mindegyik helye, ahol van obj az 1, ahol nincs 0.
        //hol nincsen 0?
        
        int[][] res = new int[2][2];
        int idx = 0;
        
        for( x=0; x<4; x++ ) {
            for( y=0; y<5; y++ ) {
                if( array[x][y] == 0 ) {
                    res[idx][0] = x;
                    res[idx++][1] = y;
                }
            }
        }
        
        objects[10] = getPos(res[0][0], res[0][1]);
        objects[11] = getPos(res[1][0], res[1][1]);
    }

            else System.out.print("0");
            if( i%sep == 0 ) System.out.print(" ");
        }
