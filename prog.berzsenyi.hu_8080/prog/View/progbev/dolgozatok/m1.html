program index;
var m, h, bmi: real;
begin
    writeln('ird be a tomeged: ');
    readln(m);
    writeln('ird be a magassagod: ');
    readln(h);        
    bmi := m/(h*h);
    if bmi > 30 
    then writeln('tulsulyos vagy!')
    else if bmi >25 
        then writeln('elhizasban szenvedsz!')
        else if bmi > 18
            then writeln ('szerencsed van testomegindexed normalis')
            else writeln('egyel tobbet! korosan sovany vagy.');
end.
