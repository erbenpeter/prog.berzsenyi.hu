program f222;


var sum:real; i:integer; //osszeg,ciklusvaltozo

begin
sum:=49+50; //legfelso, legalso gomb sugara (a ket legnagyobb sejtes szerint)

for i:=30 to 48 do begin
sum:=sum+sqrt(200)*sqrt(2*i+2-50); //2-nkent a kulonbseg (prs-prs, ptln-ptln)
end;

sum:=sum+sqrt(200)*sqrt(31+30-50); //ket legkisebb egymas mellett van

sum:=sum*1000; //legyen mikrometer a milimeterbol
writeln(round(sum)); //ezt egeszre kerekitve irjuk ki!

end.
