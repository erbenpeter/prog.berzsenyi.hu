using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

namespace Kontener
{
    class Container
    {
        public int weight;
        public bool ready = false; // kész van, tehát rá többen nem lehet pakolni
        public bool movable = true; // mozgatható-e

        public Container( int _weight )
        {
            weight = _weight;
        }
    }

    class Program
    {
        // n->konténerek, k->max súly
        private int n, k;
        private List<Container> containers;

        public Program()
        {
            StreamReader ins = new StreamReader("kontener.be");
            string[] line = ins.ReadLine().Split(' ');
            n = int.Parse( line[0] );
            k = int.Parse( line[1] );

            containers = new List<Container>( n );
            line = ins.ReadLine().Split(' ');            
            for (int i = 0; i < n; i++)
            {
                containers.Add( new Container( int.Parse( line[i] ) ) );
            }

            doSomeMagic();
        }

        private void doSomeMagic()
        {
            int maxWeight;
            int idx, idx2;
            int startCount = containers.Count;
            while (true)
            {
                // kiválasztjuk jobbról a legnagyobbat, aki még nincs kész!
                idx = -1;
                maxWeight = 0;
                for (int i = 0; i < containers.Count; i++)
                {
                    if (!containers[i].ready && maxWeight <= containers[i].weight)
                    {
                        idx = i;
                        maxWeight = containers[i].weight;
                    }
                }
                if (idx == -1) break; // vége van!

                idx2 = -1;
                maxWeight = 0;
                for (int i = 0; i < idx; i++)
                {
                    // tőle balra lévő itemeket megnézzük, hogy melyiket tudjuk rádobni
                    if (containers[i].movable && maxWeight <= containers[i].weight && containers[i].weight < containers[idx].weight)
                    {
                        // ha még mozgatható, és rátehető a legnagyobb csomagra
                        idx2 = i;
                        maxWeight = containers[i].weight;
                    }
                }
                // idx2 = az idx-től balra levő legközelebbi legnagyobb még rátehető doboz.
                if (idx2 == -1)
                {
                    containers[idx].ready = true; // nem lehet már rá többet pakolni
                }
                else
                {
                    containers[idx].movable = false;
                    containers[idx].weight = containers[idx2].weight; // rátesszük
                    containers.RemoveAt(idx2);
                }
            }

            StreamWriter outs = new StreamWriter("kontener.ki");
            outs.WriteLine( startCount - containers.Count );
            outs.Close();
        }

        static void Main(string[] args)
        {
            new Program();
        }
    }
}
