1. feladat megoldása

program f1;
{3-ra végződő egész számok összege 1..1000}

var x,s : longint;
begin
    x := 3; s := 0;
    while x <= 1000 do
    begin
        s := s + x;
        x := x + 10;
    end;
    writeln(s);
end.

Második módszer

program f1b;
{3-ra végződő egész számok összege 1..1000}

var x,s : longint;
begin
    s := 0;
    for x := 1 to 1000 do
    begin
        if (x mod 10) = 3
            then s := s + x;            
    end;
    writeln(s);
end.
