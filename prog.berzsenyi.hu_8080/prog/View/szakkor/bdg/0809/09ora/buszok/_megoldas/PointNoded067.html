package hu.krivan.szakkor.dijkstra;

/**
 *
 * @author Bálint Kriván
 */
public class PointNode extends Node {

    public int x, y;

    public PointNode( String input )
    {
        String[] tmp = input.split(" ");
        x = Integer.valueOf(tmp[0]);
        y = Integer.valueOf(tmp[1]);
    }

    public PointNode(int x, int y) {
        this.x = x;
        this.y = y;
    }

    @Override
    public int getDistanceFrom(PointNode o) {
        return Math.abs(this.x - o.x) + Math.abs(this.y - o.y);
    }

    @Override
    public int getDistanceFrom(RouteNode route) {
        return route.getDistanceFrom(this);
    }

    @Override
    public String toString() {
        return new String("("+x+","+y+")");
    }

}
