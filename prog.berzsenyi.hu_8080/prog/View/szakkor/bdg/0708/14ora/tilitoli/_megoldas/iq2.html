program iq2;

{$IFDEF WINDOWS}
  {$INFO Windows Version}
{$ELSE}
  {$INFO Non-Windows Version}
{$ENDIF}

{$IFDEF WINDOWS}
uses windows;
{$ENDIF}

//0hwppppp
const c_w1  = $0;
      c_w2  = $20;
      c_h1  = $0;
      c_h2  = $40;

      c_pos = $1F;      
      c_w   = c_w2;
      c_h   = c_h2;
      c_col = c_w or c_h;

      c_yellow  = c_w2 or c_h2;
      c_black   = c_w1 or c_h2;
      c_gray    = c_w2 or c_h1;
      c_red     = c_w1 or c_h1;

      c_empty   = $FF;
      
const N = 10;
      M = 20;
      defvoltsize = 2000;
      qsize       = 10000;

const L : array[0..3] of shortint = (-4,4,-1,1);

type  TElem   = byte;
      TState  = array[1..N]of TElem;
      TTabla  = array[0..M-1] of byte;
      TStates = array of TState;

      PCella  = ^TCella;
      TCella  = record
                  s:TState;
                  apa:PCella;
                end;

var start   : TState;

    voltt   : TStates;
    voltdb,
    voltsize: longint;
    
    que     : array[1..qsize] of PCella;
    eleje,
    vege    :longint;
    
    counter : longint;
    {$IFDEF WINDOWS}
    starttime:longword;
    {$ENDIF}

procedure dump(s:TState); forward;

function gw(e:TElem):byte;
begin
  result:=byte((e and c_w) shr 5);
end;

function gh(e:TElem):byte;
begin
  result:=byte((e and c_h) shr 6);
end;

procedure sp(var e:TElem;x:byte);
begin
  e:=(e and c_col) or x;
end;

function upack(s:TState;var t:TTabla):boolean;
var i:byte;
    uj:byte;
    x,y:byte;
begin
  for i:=0 to M-1 do t[i]:=c_empty;
  
  result:=false;
  
  for i:=1 to N do
    begin
      for x:=0 to gw(s[i]) do
        for y:=0 to gh(s[i]) do
          begin
            uj:=byte( byte(s[i] and c_pos)+byte(x)+byte(y*4) );
            if t[uj]=c_empty then
              t[uj]:=(s[i] and c_col)
            else
              exit;
          end;
    end;
    
  result:=true;
end;

function lep(var s:TState;i:byte;d:shortint):boolean;
var poz:byte;
    t:TTabla;
begin
  result:=true;
  
  poz:=s[i] and c_pos;
  
  case d of
    +1:if (poz mod 4)+gw(s[i]) > 2 then result:=false;
    -1:if (poz mod 4)=0            then result:=false;
    +4:if (poz div 4)+gh(s[i]) > 3 then result:=false;
    -4:if (poz div 4)=0            then result:=false;
  end;
  
  if result then
    begin
      sp(s[i],byte((s[i] and c_pos)+d));
      result:=upack(s,t);
    end;
end;

function gt(a,b:TState):boolean;
var i:byte;
    at,bt:TTabla;
begin
  upack(a,at);
  upack(b,bt);
  
  i:=0;
  while(i<M)and(at[i]=bt[i])do inc(i);
  if i=M then gt:=false else gt:=(at[i] >= bt[i]);
end;


function volt(s:TState):boolean;
var j:longint;
    e,u,k:longint;
begin
  //bináris keresés
  e:=1;
  u:=voltdb;
  result:=false;
  while(e<=u)and not(result) do
    begin
      k:=(e+u) div 2;
      if gt(s,voltt[k]) then e:=k+1
      else if gt(voltt[k],s) then u:=k-1
      else result:=true;      
    end; 
  //---------------  

  //"e" helyre kell tenni
  if not(result) then
    begin
      if voltdb = voltsize then
        begin
          voltsize:=voltsize+100;
          setlength(voltt,voltsize+1);
        end;
      
      for j:=voltdb downto e do voltt[j+1]:=voltt[j];
      voltt[e]:=s;
      
      inc(voltdb);
    end;
end;        

procedure qpush(p:PCella);
begin
  inc(vege);
  if vege > qsize then vege:=1;
  que[vege]:=p;
end;

procedure qpop(var p:PCella);
begin
  p:=que[eleje];  
  inc(eleje);
  if eleje > qsize then eleje:=1;
end;

function qempty:boolean;
begin
  qempty := (vege+1=eleje) or (eleje=1)and(vege=qsize);
end;

procedure print(p:PCella);
begin
  inc(counter);
  if p^.apa<>nil then print(p^.apa);
  dump(p^.s);
end;

procedure kiir(p:PCella);
{$IFDEF WINDOWS}
var endtime:longword;
{$ENDIF}
begin
  {$IFDEF WINDOWS}
  endtime:=gettickcount;
  {$ENDIF}
    
  counter:=0;
  print(p);
  writeln(counter-1,' le''pe''s');
  {$IFDEF WINDOWS}
  writeln(endtime-starttime,' ms');
  {$ENDIF}  
  
  halt;
end;

procedure megold;
var s:TState;
    i,j:byte;
    p,up:PCella;
begin
  new(p);
  p^.s:=start;
  p^.apa:=nil;
    
  qpush(p);
  while(not(qempty))do
    begin
      qpop(p);
      
      for i:=1 to N do
        for j:=0 to 3 do
          begin
            s:=p^.s;
            if lep(s,i,L[j]) and not(volt(s)) then
              begin                
                new(up);
                up^.s:=s;
                up^.apa:=p;
                
                qpush(up);
                
                if (s[1] and c_pos)=13 then kiir(up);
              end;            
          end;
    end;
end;

procedure dump(s:TState);
var t:TTabla;
    ch:Char;
    i:byte;
begin
  upack(s,t);
  
  writeln('+----+');
  for i:=0 to M-1 do
    begin
      if i mod 4 = 0 then write('|');
      case t[i] of
        c_empty:ch:=' ';
        c_yellow:ch:='Y';
        c_black:ch:='B';
        c_red:ch:='R';
        c_gray:ch:='G';
        else ch:='?';
      end;
      write(ch);
      if i mod 4 = 3 then writeln('|');
    end;
  writeln('+-  -+');
end;

procedure load;
begin
  voltdb:=0;
  voltsize:=defvoltsize;
  setlength(voltt,voltsize+1);
  
  eleje:=1;
  vege:=qsize;
  
  start[1] := 1   or c_yellow;
  start[2] := 0   or c_black;
  start[3] := 3   or c_black;
  start[4] := 12  or c_black;
  start[5] := 15  or c_black;
  start[6] := 13  or c_red;
  start[7] := 14  or c_red;
  start[8] := 17  or c_red;
  start[9] := 18  or c_red;
  start[10]:= 9   or c_gray;
  
  {start[1] := 12  or c_yellow;
  start[2] := 0   or c_black;
  start[3] := 1   or c_black;
  start[4] := 3   or c_black;
  start[5] := 15  or c_black;
  start[6] := 2   or c_red;
  start[7] := 6   or c_red;
  start[8] := 10  or c_red;
  start[9] := 11  or c_red;
  start[10]:= 8   or c_gray;}    
end;

begin
  load;
 
  {$IFDEF WINDOWS}
  starttime:=gettickcount;
  {$ENDIF}
  megold;
end.

{
function eq(a,b:TState):boolean;
var at,bt:TTabla;
    i:byte;
begin
  upack(a,at);
  upack(b,bt);
  
  result:=false;
  for i:=0 to M-1 do if at[i]<>bt[i] then exit;
  result:=true;  
end;
}
