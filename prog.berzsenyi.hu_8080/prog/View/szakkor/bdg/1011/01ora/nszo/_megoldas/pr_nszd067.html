program osze;

var t:text;
a:array[1..101] of array[0..5] of int64;
i,j,k:longint; b:string[10]; hib:integer;
s,m:ansistring;

begin
assign(t,'szam.txt'); reset(t);

for j:=1 to 100 do begin

for i:=1 to 4 do begin read(t,b); val(b,a[j][i],hib); end;
readln(t,b); val(b,a[j][5],hib);

end;

for k:=5 downto 1 do begin

for i:=1 to 100 do begin
a[101][k]:=a[i][k]+a[101][k];
end;
a[101][k-1]:=(a[101][k] div 10000000000);
a[101][k]:=(a[101][k] mod 10000000000);
end;

s:='';

for i:=0 to 5 do begin
str(a[101][i]:10,m);
s:=s+m;
end;
i:=1;
while (i<=length(s)) and (s[i]=' ') do inc(i);
delete(s,1,i-1);

for i:=1 to length(s) do begin if s[i]=' ' then s[i]:='0'; end;

m:=copy(s,1,10);

writeln(m);

end.
