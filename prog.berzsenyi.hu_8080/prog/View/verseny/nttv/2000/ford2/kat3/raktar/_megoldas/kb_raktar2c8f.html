using System;
using System.Collections.Generic;
using System.Text;

namespace Raktar
{
    class Coords
    {
        public int x, y;

        public Coords(int _x, int _y)
        {
            x = _x;
            y = _y;
        }
    }

    class Program
    {
        private int n, m;
        private int[,] table;
        private int[,] steps;
        private Queue<Coords> q;
        private Queue<Coords> deadEnds = new Queue<Coords>();

        public Program()        
        {
            System.IO.StreamReader ins = new System.IO.StreamReader("raktar7.be");
            System.IO.StreamWriter outs = new System.IO.StreamWriter("raktar7.ki");
            string[] fLine = ins.ReadLine().Split(' ');
            n = int.Parse(fLine[0]);
            m = int.Parse(fLine[1]);
            string line;
            table = new int[n,m];
            steps = new int[n, m];

            int x = -1, y = -1;

            for (int j = 0; j < m; j++)
            {
                line = ins.ReadLine(); // n hosszú!
                for (int i = 0; i < n; i++)
                {
                    if (x == -1 && line[i] == '.')
                    {
                        // megtaláltuk az első szabad mezőt, elmentjük
                        // innen indul majd a keresés.
                        x = i; y = j;
                    }
                    table[i, j] = (line[i] == '.') ? 0 : 1;
                    steps[i, j] = -1;
                }
            }

            int maxSteps = walk( x, y, true ), maxStepCount;
            Coords c;
            while( deadEnds.Count != 0 )
            {
                c = deadEnds.Dequeue();
                clearSteps();
                maxStepCount = walk( c.x, c.y, false );
                if (maxStepCount > maxSteps)
                {
                    maxSteps = maxStepCount;
                }
            }
            outs.WriteLine(maxSteps);
            outs.Close();
        }

        private void clearSteps()
        {
            for (int j = 0; j < m; j++)
            {
                for (int i = 0; i < n; i++)
                {
                    steps[i, j] = -1;
                }
            }
        }

        private int walk(int x, int y, bool markDeadEnds)
        {
            int maxSteps = 0;
            steps[x, y] = 0;
            q = new Queue<Coords>();
            q.Enqueue( new Coords( x, y ) );
            Coords c;
            bool e, f, g, h;
            while (q.Count != 0)
            {
                c = q.Dequeue();
                // kivesszük a koordinátát, és megnézzük, hogy merre mehetünk.
                e = check(c.x + 1, c.y, c);
                f = check(c.x, c.y + 1, c);
                g = check(c.x - 1, c.y, c);
                h = check(c.x, c.y - 1, c);
                if (!e && !f && !g && !h)
                {
                    // zsákutca!
                    if (markDeadEnds) deadEnds.Enqueue(c);
                    // zsákutcában lehet maximális távolság!
                    if (maxSteps < steps[c.x, c.y])
                    {
                        maxSteps = steps[c.x, c.y];
                    }
                } 
            }

            return maxSteps;
        }

        private bool check(int x, int y, Coords c)
        {
            if (x < n && y < m && x >= 0 && y >= 0)
            {
                if (table[x, y] == 0 && steps[x, y] == -1)
                {
                    // léphetünk oda és még nem jártunk ott.
                    steps[x, y] = steps[c.x, c.y]+1;
                    q.Enqueue( new Coords(x, y) );
                    return true;
                }
            }
            return false;
        }

        static void Main(string[] args)
        {
            new Program();
        }
    }
}
