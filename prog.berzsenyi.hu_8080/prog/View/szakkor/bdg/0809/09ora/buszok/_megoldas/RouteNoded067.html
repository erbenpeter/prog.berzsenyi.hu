package hu.krivan.szakkor.dijkstra;

import java.util.ArrayList;

/**
 *
 * @author Bálint Kriván
 */
public class RouteNode extends Node {

    /**
     * Forduló pontok száma
     */
    public int n;
    /**
     * Jegy ára
     */
    private int cost;
    private ArrayList<Point> turningPoints;

    private class Point {

        /**
         * Koordináták
         */
        public int x,  y;

        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    private class Section {

        public int x1,  x2,  y1,  y2;

        public Section(Point a, Point b) {
            x1 = a.x;
            y1 = a.y;
            x2 = b.x;
            y2 = b.y;
            int z;
            if (x2 < x1) {
                z = x1;
                x1 = x2;
                x2 = z;
            }
            if (y2 < y1) {
                z = y1;
                y1 = y2;
                y2 = z;
            }
        }

        /**
         * A szakasz és egy pont távolsága
         * 
         * @param x a pont x koordinátája
         * @param y a pont y koordinátája
         * @return távolság
         */
        public int getDistanceFrom(int x, int y) {
            int d = 0;
            if (y2 == y1) {
                // a szakasz párhuzamos az x tengellyel, kérdéses, hogy a (p.x, y1) pont rajta van-e a szakaszon?
                d = Math.abs(y - y1);
                if (x < x1) {
                    d += x1 - x;
                } else if (x2 < x) {
                    d += x - x2;
                }
            } else if (x2 == x1) {
                // a szakasz merőleges az x tengelyre.
                d = Math.abs(x - x1);
                if (y < y1) {
                    d += y1 - y;
                } else if (y2 < y) {
                    d += y - y2;
                }
            }
            return d;
        }

        public int getDistanceFrom(Section o) {
            int d = 0;
            if (this.x1 == this.x2 && o.x1 == o.x2) {
                // a két szakasz párhuzamos, és mindkettő merőleges az x tengelyre.
                d = Math.abs(this.x1 - o.x1);
                if (this.y2 < o.y1) {
                    d += o.y1 - this.y2;
                } else if (o.y2 < this.y1) {
                    d += this.y1 - o.y2;
                }
            } else if (this.y1 == this.y2 && o.y1 == o.y2) {
                // a két szakasz párhuzamos, és mindkettő párhuzamos az x tengellyel.
                d = Math.abs(this.y1 - o.y1);
                if (this.x2 < o.x1) {
                    d += o.x1 - this.x2;
                } else if (o.x2 < this.x1) {
                    d += this.x1 - o.x2;
                }
            } else {
                // a két szakasz egyenese metszi egymást.
                // keressük a metszéspontot
                int x, y;
                if (this.x1 == this.x2) {
                    x = x1;
                    y = o.y1;
                } else {
                    x = o.x1;
                    y = y1;
                }
                // nyílván a két szakasz távolsága, a két szakasz és a metszéspont távolságösszege.
                d = this.getDistanceFrom(x, y) + o.getDistanceFrom(x, y);
            }
            return d;
        }
    }

    public RouteNode(String readLine) {
        String[] tmp = readLine.split(" ");
        n = Integer.valueOf(tmp[0]);
        cost = Integer.valueOf(tmp[1]);

        int x, y;
        turningPoints = new ArrayList<Point>();
        for (int i = 2; i < tmp.length; i += 2) {
            x = Integer.valueOf(tmp[i]);
            y = Integer.valueOf(tmp[i + 1]);
            turningPoints.add(new Point(x, y));
        }
    }

    @Override
    public int getDistanceFrom(PointNode point) {
        int d, min = Integer.MAX_VALUE;
        Section section;
        for (int i = 0; i < n; i++) {
            section = new Section(turningPoints.get(i), turningPoints.get((i + 1) % n));
            d = section.getDistanceFrom(point.x, point.y);
            if (d < min) {
                min = d;
            }
        }
        return min;
    }

    @Override
    public int getDistanceFrom(RouteNode o) {
        Section section1, section2;
        int d, min = Integer.MAX_VALUE;
        for (int a = 0; a < this.n; a++) {
            section1 = new Section(turningPoints.get(a), turningPoints.get((a + 1) % n));
            for (int b = 0; b < o.n; b++) {
                section2 = new Section(o.turningPoints.get(b), o.turningPoints.get((b + 1) % o.n));
                d = section1.getDistanceFrom(section2);
                if (d < min) {
                    min = d;
                }
            }
        }

        return min;
    }

    @Override
    public String toString() {
        return new String("BUS");
    }

}
