#include <stdio.h>

char ismer[100][100];
int hivva[100];
int n;

void nullaz()
{
    int i, j;

    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)
            ismer[i][j] = 0;
}

int beolvas()
{
    int i, j;

    if (scanf("%d", &n) != 1) {
        printf("hibas bemenet (nincs n)\n");
        return 0;
    }

    for (i = 0; i < n; i++) {
        while (1) {
            if (scanf("%d", &j) != 1) {
                printf("hibas bemenet (nincs ismeros)\n");
                return 0;
            }

            if (j == 0)
                break;

            ismer[i][j - 1] = 1;
        }
    }

    return 1;
}

int rek(int i)
{
    int j, s = 0;

    if (hivva[i])
        return 0;

    hivva[i] = 1;

    for (j = 0; j < n; j++)
        if (ismer[i][j])
            s += rek(j);

    return s + 1;
}

int hiv(int i)
{
    int j;

    for (j = 0; j < n; j++)
        hivva[j] = 0;

    return rek(i);
}

int main()
{
    int i, m, max = -1, max_ind = -1;

    nullaz();

    if (!beolvas())
        return 1;

    for (i = 0; i < n; i++) {
        m = hiv(i);
        if (m > max) {
            max = m;
            max_ind = i;
        }
    }

    printf("%d\n", max_ind + 1);

    return 0;
}
