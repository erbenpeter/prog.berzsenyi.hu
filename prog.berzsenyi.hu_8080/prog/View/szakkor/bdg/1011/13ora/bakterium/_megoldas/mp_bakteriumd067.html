Program bac;
uses graph,wincrt;

type sejt=array[0..100,0..100] of byte;

Var f:text;
    p1,p2,r1,i,j,x,y,c,r,x1,x2,y1,y2:byte;
    ido:longint;
    h1,h2:sejt;
    vege:byte;
    gd,gm:integer;

Procedure h0;
Begin
for i:=0 to 100 do
    Begin
    for j:=0 to 100 do
        Begin
        h1[i,j]:=0;
        End;
    End;
for i:=0 to 100 do
    Begin
    for j:=0 to 100 do
        Begin
        h2[i,j]:=0;
        End;
    End;
End;

Procedure beolvas;
Begin
readln(f,c);
for i:=1 to c do
    Begin
    readln(f,x1,y1,x2,y2);
        for x:=x1 to x2 do
            Begin
            for y:=y1 to y2 do
                Begin
                h1[x,y]:=1;
                End;
            End;
    End;
End;

Procedure szimulal(var q1,q2:sejt);
Begin
vege:=1;
for i:=1 to 100 do
    Begin
    for j:=1 to 100 do
        Begin
        if q1[i,j]=1 then vege:=0;
        End;
    End;
if vege=0 then
    Begin
    for i:=1 to 100 do
        Begin
        for j:=1 to 100 do
            Begin
            q2[i,j]:=q1[i,j];
            End;
        End;
    for i:=1 to 100 do
        Begin
        for j:=1 to 100 do
            Begin
            if (q1[i,j-1]=1) and (q1[i-1,j]=1) then q2[i,j]:=1;
            if (q1[i,j-1]=0) and (q1[i-1,j]=0) then q2[i,j]:=0;
            End;
        End;
    ido:=ido+1;       
    End;
End;

Procedure initgraf;
Begin
detectgraph(gd,gm);
initgraph(gd,gm,'');
End;

Procedure kockap(x0,y0:byte);
Begin
setfillstyle(SolidFill,red);
bar(x0*8,y0*8,(x0+1)*8,(y0+1)*8);
End;

Procedure kockaf(x0,y0:byte);
Begin
setfillstyle(SolidFill,black);
bar(x0*8,y0*8,(x0+1)*8,(y0+1)*8);
End;

Procedure szimulacio(o1,o2:sejt);
Begin
    for p1:=1 to 100 do
        Begin
        for p2:=1 to 100 do
            Begin
            if o1[p1,p2]=1 then kockap(p1,p2);
            if (o1[p1,p2]=0) and (o2[p1,p2]=1) then kockaf(p1,p2);
            End;
        End;
End;

Begin
initgraf;
assign(f,'bacS.in');
reset(f);
readln(f,r);
for r1:=1 to r do
    Begin
    h0;
    beolvas;
    vege:=0;
    ido:=0;
    while vege<>1 do
        Begin
        if ido mod 2=0 then
            Begin
            szimulacio(h1,h2);
            szimulal(h1,h2);   
            End
            else
            Begin
            szimulacio(h2,h1);
            szimulal(h2,h1);
            End;
        delay(10);   
        End;
    writeln(ido);
    End;
closegraph;
close(f);   
End.
