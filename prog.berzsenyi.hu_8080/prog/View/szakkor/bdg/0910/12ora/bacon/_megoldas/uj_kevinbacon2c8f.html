/*
    Informatika: algoritmus szakkor
    Feladat: Kevin Bacon jatek
    Forras: http://prog.berzsenyi.hu:8080/prog/View/szakkor/bdg/0910/12ora/bacon
    Uray M. Janos, 2009. december 8.
*/
#include <stdio.h>

#define MaxActors 200
#define MaxFilms 2000
#define MaxPlays 2000
#define MaxFilmLen 128
#define MaxActorLen 32


// Szinesek, filmek es szereplesek tarolasa
int ActorNum, FilmNum, PlayNum;
char Actor[MaxActors][MaxActorLen];
char Film[MaxFilms][MaxFilmLen];
struct sPlay {
    int Actor, Film;
} Play[MaxPlays];
// A szineszek elszomszedsagi matrixa
int Edge[MaxActors][MaxActors];
// A csucsokrol eltarolt informaciok
struct sNode {
    // A kiindulastol valo tavolsag
    int Dist;
    // Szulo (amelyik csucsnak meg nincs szuloje, annak ez -1; kezdoelem szuloje onmaga)
    int Parent;
} Node[MaxActors];
// A sor adatszerkezet elemei
int FIFO[MaxActors];
// A sor eleje es vege
int FOut, FIn;

bool ReadActors (const char *);
bool ReadFilms (const char *);
bool ReadPlays (const char *);

void CutNewLineChar (char *);
bool Match (const char *, const char *);

void BuildGraph (void);
void BreadthFirstSearch (int);

bool GetActors (int &, int &);
void PrintShortestRoute (int, int);

int main (int AN, char * Args []) {
    int A, B;
    
    // Parameterek ellenorzese, allomanyok beolvasasa
    if (AN < 4)
        return 1;
    if (!ReadActors(Args[1]))
        return 1;
    if (!ReadFilms(Args[2]))
        return 1;
    if (!ReadPlays(Args[3]))
        return 1;
    
    // Graf felepitese
    BuildGraph();
    
    // Szineszek bekerese, ut kiirasa
    if (GetActors(A, B))
        PrintShortestRoute(A, B);
    return 0;
}

// Szineszek beolvasasa
bool ReadActors (const char * FN) {
    FILE * F = fopen(FN, "r");
    if (F == NULL)
        return false;
    fscanf(F, "%d\n", &ActorNum);
    for (int I = 0; I < ActorNum; I++) {
        fgets(Actor[I], MaxActorLen, F);
        CutNewLineChar(Actor[I]);
    }
    fclose(F);
    return true;
}
// Filmek beolvasasa
bool ReadFilms (const char * FN) {
    FILE * F = fopen(FN, "r");
    if (F == NULL)
        return false;
    fscanf(F, "%d\n", &FilmNum);
    for (int I = 0; I < FilmNum; I++) {
        fgets(Film[I], MaxFilmLen, F);
        CutNewLineChar(Film[I]);
    }
    fclose(F);
    return true;
}
// Szereplesek beolvasasa
bool ReadPlays (const char * FN) {
    FILE * F = fopen(FN, "r");
    if (F == NULL)
        return false;
    fscanf(F, "%d\n", &PlayNum);
    for (int I = 0; I < PlayNum; I++) {
        fscanf(F, "%d %d\n", &Play[I].Actor, &Play[I].Film);
        Play[I].Actor--;
        Play[I].Film--;
    }
    fclose(F);
    return true;
}
// Az fgets fuggveny altal beolvassott ujsorkaraktert levagasa
void CutNewLineChar (char * Str) {
    for (int I = 0; Str[I]; I++)
        if (Str[I] == '\n' || Str[I] == '\r') {
            Str[I] = 0;
            return;
        }
}
// Eldonti ket stringrol, hogy egyenloek-e
bool Match (const char * A, const char * B) {
    int I;
    for (I = 0; A[I]; I++)
        if (A[I] != B[I])
            return false;
    return B[I] == 0;
}

// A graf felepitese a beolvasott adatokbol
void BuildGraph (void) {
    int I, J;
    // Matrix nullazasa
    for (I = 0; I < ActorNum; I++)
        for (J = 0; J < ActorNum; J++)
            Edge[I][J] = -1;
    // Minden szerepre megnezi, hogy az elozoek kozott van-e olyan, amellyel azonos a film.
    // Ha van, akkor a ket szereples szineszei szomszedok.
    for (I = 0; I < PlayNum; I++)
        for (J = 0; J < I; J++)
            if (Play[I].Film == Play[J].Film) {
                Edge[Play[I].Actor][Play[J].Actor] = Play[I].Film;
                Edge[Play[J].Actor][Play[I].Actor] = Play[I].Film;
            }
}

// A szelessegi bejaras
// Start: a kezdoelem
void BreadthFirstSearch (int Start) {
    int I;
    FIn = FOut = 0;
    for (I = 0; I < ActorNum; I++) {
        Node[I].Dist = -1;
        Node[I].Parent = -1;
    }
    Node[Start].Dist = 0;
    Node[Start].Parent = Start;
    FIFO[FIn++] = Start;
    do {
        for (I = 0; I < ActorNum; I++)
            if (Edge[FIFO[FOut]][I] != -1 && Node[I].Parent == -1) {
                FIFO[FIn++] = I;
                Node[I].Dist = Node[FIFO[FOut]].Dist + 1;
                Node[I].Parent = FIFO[FOut];
            }
        FOut++;
    } while (FIn > FOut);
}

// Beker ket szinesznevet, es konvertalja sorszamma
bool GetActors (int & N1, int & N2) {
    char NameA[MaxActorLen], NameB[MaxActorLen];
    
    printf("Irjon be ket szineszt kulon sorba:\n");
    fgets(NameA, MaxActorLen, stdin);
    fgets(NameB, MaxActorLen, stdin);
    CutNewLineChar(NameA);
    CutNewLineChar(NameB);
    for (N1 = 0; N1 < ActorNum; N1++)
        if (Match(Actor[N1], NameA))
            break;
    for (N2 = 0; N2 < ActorNum; N2++)
        if (Match(Actor[N2], NameB))
            break;
    if (N1 == ActorNum) {
        printf("Az elso szinesz neve hibas!\n");
        return false;
    }
    if (N2 == ActorNum) {
        printf("A masodik szinesz neve hibas!\n");
        return false;
    }
    return true;
}

// Kiirja a legrovidebb utat ket szinesz kozott
void PrintShortestRoute (int N1, int N2) {
    int X = N2;
    BreadthFirstSearch(N1);
    if (Node[X].Dist == -1) {
        printf("Infinite.\n");
        return;
    }
    printf("%s", Actor[X]);
    while (Node[X].Parent != X) {
        printf(" -> %s", Film[Edge[X][Node[X].Parent]]);
        X = Node[X].Parent;
        printf(" -> %s", Actor[X]);
    }
    printf("\n");
}
