{$MODE OBJFPC}
program kruskalprog;

const INF = maxlongint;

type  Tapa    = array of longint;
      TStack  = array of longint;
      TEl     = record
                  x,y:longint;
                  k:longint;
                  ok:boolean;
                end;
      TKupac  = array of TEl;

var   N,M:longint;
      apa:TApa;
      E:TKupac;
      sum:longint;
      stack:TStack;
      ssize:longint;
                
procedure swap(var a,b:TEl);
var c:TEl;
begin
  c:=a;
  a:=b;
  b:=c;
end;

function gt(a,b:TEl):boolean; inline;
begin
  gt:=(a.k>b.k)
end;

procedure kupacol(T:TKupac;N:longint;k:longint);
var E:TEl;
    apa,fiu:longint;
begin
  E:=T[k];
  
  apa:=k;
  fiu:=2*apa;
  if (fiu<N)and(gt(T[fiu+1],T[fiu])) then inc(fiu);

  while(fiu<=N)and(gt(T[fiu],E)) do
    begin
      T[apa]:=T[fiu];
      
      apa:=fiu;
      fiu:=2*apa;
      if (fiu<N)and(gt(T[fiu+1],T[fiu])) then inc(fiu);
    end;
  
  T[apa]:=E;    
end;

procedure epit(T:TKupac;N:longint);
var i:longint;
begin
  for i:=(N div 2) downto 1 do kupacol(T,N,i);
end;


procedure hsort(T:TKupac;N:longint);
var i:longint;
begin
  epit(T,N);
  
  for i:=1 to N-1 do
    begin
      swap(T[1],T[N-i+1]);
      kupacol(T,N-i,1);
    end;
end;

function find(x:longint):longint;
begin
  if apa[x]>0 then
    begin
      result:=find(apa[x]);
      apa[x]:=result;      
    end
  else
    begin
      result:=x;
    end;
end;

//kisebbet a nagyobb alá
procedure link(x,y:longint); inline;
begin
  if -apa[x] < -apa[y] then
    begin
      apa[y]:=apa[x]+apa[y];
      apa[x]:=y;
    end
  else
    begin
      apa[x]:=apa[x]+apa[y];
      apa[y]:=x;
    end;
end;

procedure union(x,y:longint);
begin
  link(find(x),find(y));
end;

procedure spush(x:longint);
begin
  inc(ssize);
  stack[ssize]:=x;
end;

procedure load;
var i:longint;
begin
  readln(N,M);
  
  setlength(E,M+1);
  for i:=1 to M do
    begin
      readln(e[i].x,e[i].y,e[i].k);
      e[i].ok:=true;
    end;

  hsort(e,M);
  setlength(apa,N+1);
  
  setlength(stack,N);
  ssize:=0;          
end;

procedure kruskal(b:boolean = true);
var i:longint;
begin
  for i:=1 to N do apa[i]:=-1;
  sum:=0;
  
  for i:=1 to M do
    begin
      if e[i].ok and (find(e[i].x)<>find(e[i].y)) then
        begin
          union(e[i].x,e[i].y);
          
          if b then spush(i);
          sum:=sum+e[i].k;
        end;
    end;
end;

var db,c,i:longint;
    minsum,min:longint;
begin
  readln(db);
  for c:=1 to db do
    begin
      load;
      
      kruskal;
      min:=sum;
      minsum:=INF;
      
      for i:=1 to N-1 do
        begin
          e[stack[i]].ok:=false;
          kruskal(false);
          if sum<minsum then minsum:=sum;
          e[stack[i]].ok:=true;
        end;
            
      writeln(min,' ',minsum);
    end;  
end.of.program
