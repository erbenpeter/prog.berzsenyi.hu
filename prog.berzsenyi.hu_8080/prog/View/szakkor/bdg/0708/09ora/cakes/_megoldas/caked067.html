using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Collections;

namespace Cookies
{
   class Program
   {
       #region Lényegi rész

       private static Megoldas Munka(Problem p)
       {
           //inicializálás

           int szum=0;
           for (int i=0; i!=p.n; i++) szum+=p.idok[i];

           int[,] choice = new int[szum + 1, szum + 1];

           BitArray lehet = new BitArray((szum+1) * (szum+1), false); //lehet 0 is, ezért +1
           lehet[0] = true;
           int s = 0;
           int t = 0;
           for (int i = 0; i != p.n; i++)
           {
               s += p.idok[i];
               t = p.idok[i];
               for (int j = szum; j != -1; j--)
               {
                   for (int k = szum; k != -1; k--)
                   {
                       if ((j >= t) && (lehet[(j - t) * (szum + 1) + k] == true))
                       {
                           lehet[j * (szum + 1) + k] = true;
                           choice[j, k] = i;
                       }
                       if ((k >= t) && (lehet[j * (szum + 1) + (k - t)] == true))
                       {
                           lehet[j * (szum + 1) + k] = true;
                           choice[j, k] = i;
                       }
                   }
               }
           }


           //optimum kikeresés
           int opt = szum;
           int opt2 = 0;
           int mj = 0;
           int mk = 0;
           for (int j = 0; j != szum+1; j++)
           {
               for (int k = 0; k != szum+1; k++)
               {
                   if (lehet[j * (szum + 1) + k])
                   {
                       opt2=Math.Max(j, Math.Max(k, szum - j - k));
                       if (opt2 < opt)
                       {
                           opt = opt2;
                           mj = j;
                           mk = k;
                       }
                   }
               }
           }

           Megoldas mo = new Megoldas(p);
           mo.osszido = opt;
           //choiceból kiolvasás
           while(true)
           {
               if ((mj == 0) && (mk == 0)) break;
               int i=choice[mj, mk];
               t = p.idok[i];
               if ((mj >= t) && (lehet[(mj - t) * (szum + 1) + mk]))
               {
                   mo.idok[0, i] = true;
                   mo.idok[2, i] = false;
                   mj -= t;
               }
               else if ((mk >= t) && (lehet[mj * (szum + 1) + (mk - t)] == true))
               {
                   mo.idok[1, i] = true;
                   mo.idok[2, i] = false;
                   mk -= t;
               }
           }

           return mo;
       }

       #endregion

       #region Vezérlés
       static List<Problem> problems = new List<Problem>();
       public class Problem
       {
           public int n;
           public int[] idok;

           public Problem(int n)
           {
               this.n = n;
               idok = new int[n];
           }
       }

       public class Megoldas
       {
           public Problem p;
           public int osszido;
           public bool[,] idok;

           public Megoldas(Problem p)
           {
               this.p = p;
               osszido = 0;
               idok = new bool[3, p.n];
               for (int i = 0; i != p.n; i++) idok[2, i] = true; //alapból minden a harmadik sütőben.
           }
       }

       static void Main(string[] args)
       {
           Beolvas("cakes.in");
           Megoldas[] megoldasok=new Megoldas[problems.Count];
           for (int i = 0; i != problems.Count; i++) megoldasok[i] = Munka(problems[i]);
           Kiir(megoldasok);

       }

       private static void Beolvas(string inp)
       {
           TextReader tr = new StreamReader(inp);
           string[] l;
           while ((l = tr.ReadLine().Split(' '))[0] != "0")
           {
               Problem p = new Problem(Convert.ToInt32(l[0]));
               for (int i = 0; i != p.n; i++) p.idok[i] = Convert.ToInt32(l[i + 1]);
               problems.Add(p);
           }
           tr.Close();
       }

       private static void Kiir(Megoldas[] megoldasok)
       {
           TextWriter tw = new StreamWriter("cakes.out");
           Megoldas mo;
           for (int i = 0; i != megoldasok.Length; i++)
           {
               mo=megoldasok[i];
               tw.WriteLine(mo.osszido);
               for (int j = 0; j != 3; j++)
               {
                   tw.Write(" " + (j + 1) + ":");
                   for (int k = 0; k != mo.p.n; k++) if (mo.idok[j, k] == true) tw.Write(" " + mo.p.idok[k]);
                   tw.WriteLine();
               }
           }
           tw.Close();
       }
       #endregion
   }
}
