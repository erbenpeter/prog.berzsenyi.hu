using System;
using System.IO;
using System.Collections.Generic;

namespace Munka
{
    class Program
    {
        private int n, f, sum = 0;
        private int[] jobs;
        private bool[,] cache;
        
        public Program()
        {
            StreamReader ins = new StreamReader("munka.be");
            n = int.Parse(ins.ReadLine());
            
            string[] line = ins.ReadLine().Split(' ');
            jobs = new int[n+1];
            for (int i = 0; i < n; i++)
            {
                jobs[i+1] = int.Parse(line[i]);
                sum += jobs[i+1];
            }

            f = sum / 2;

            cache = new bool[f+1,n+1];

            // inicializálás -> minden hamis
            for( int i=0; i<=f; i++ )
                for( int j=0; j<=n; j++ )
                    cache[i,j] = false;         
        }

        public void doMagic()
        {
            cache[jobs[1],1] = (jobs[1] <= f); // mivel X <= F

            int i, x;
            
            for( i=2; i<=n; i++ ) {
                for( x=1; x<=f; x++ ) {
                    cache[x, i] = (x == jobs[i]) || cache[x, i-1] || (x > jobs[i] && cache[x-jobs[i], i-1]);
                }
            }

            // megkeressük a T1-et, tehát a legnagyobb X-et, melyre cache[X,N] == TRUE
            int t1 = 0;
            for( x=f; x>=1; x-- ) {
                if( cache[x, n] ) {
                    t1 = x;
                    break;
                }
            }

            // lista mely tartalmazza az összes munka indexet
            List<int> rem = new List<int>(n);
            for( int j=1; j<=n; j++ )
                rem.Add(j);
            
            StreamWriter outs = new StreamWriter("munka.ki");
            
            // nyílván max(t1,t2) = sum-t1, hiszen t1<=t2
            outs.WriteLine(sum-t1);
            
            // egy megoldást helyreállítunk
            x = t1;
            i = n;
            
            do {
                while( i>0 && cache[x,i] ) i--;
                // találtunk egy olyat, amikor már nem hozható össze az első i-ből az adott X
                // de i+1-ből már igen, tehát az tuti benne van, ezt vegyük ki (visszafele haladunk)
                x -= jobs[i+1];
                outs.Write( "{0} ", i+1 );
                rem.Remove( i+1 ); // maradékból kivesszük
            } while( x != 0 );
            outs.WriteLine();

            // kiírtuk az első processzorhoz tartozó munkákat, a maradék a második processzoré:
            for( int j=0; j<rem.Count; j++ )
                outs.Write( "{0} ", rem[j] );
            outs.WriteLine();
            
            outs.Close();
        }
        
        public static void Main(string[] args)
        {
            new Program().doMagic();            
        }
    }
}
