class Tree{
    int M;
    //int[][] adj;
    List<Integer>[] edges;
    int[]  degree;
    int[]  depth;
    int[]  parent;
    

    public Tree(int M, String sor){
       
        this.M = M;       
        degree = new int[M];
        edges = new ArrayList[M];

        for(int i=0;i<M;i++){
            degree[i] = 0;
            edges[i] = new ArrayList<Integer>();            
        }

        for(int i=0;i<M-1;i++){
            int a = Integer.parseInt(sor.split(" ")[2*i]);
            int b = Integer.parseInt(sor.split(" ")[2*i+1]);            
            degree[a]++;
            degree[b]++;
            edges[a].add(b);
            edges[b].add(a);
        }
    }

    public boolean adj(int a, int b){
        return edges[a].contains(b);
    }

    public String degreeList(){
        String s="";
        List<Integer> l = new ArrayList<Integer>();
        for(int i= 0; i<M; i++){
            l.add(degree[i]);
        }
        Collections.sort(l);
        for(int i= 0; i<M; i++){
            s += " "+l.get(i);
        }
        return s;
    }

    public Center Center(){
        Center c = new Center();
        int[] deg2 = new int[M];

        List<Integer> L1 = new ArrayList<Integer>();


        int r = M;
        for(int i = 0; i<M;i++){
            deg2[i] = degree[i];
            if(deg2[i]==1){
                L1.add(i);
                r--;
            }
        }
        while(r > 0){
            List<Integer> L2 = new ArrayList<Integer>();
            for(Iterator it = L1.iterator(); it.hasNext(); ){
                int e = (Integer)it.next();
                for(int f=0; f<M;f++){
                    if(adj(e,f)){
                        deg2[f]--;
                        if(deg2[f]==1){
                            L2.add(f);
                            r--;
                        }
                    }
                }
            }

            L1 = L2;
        }

        c.db = L1.size();
        c.c1 = L1.get(0);
        if(c.db>1){
            c.c2 = L1.get(1);
        }
        return c;
    }

    public int depth(int root){
        depth = new int[M];
        parent = new int[M];
        int[] bejart = new int[M];
        for(int i = 0; i<M;i++) {
            bejart[i] = 0;
            parent[i] = i;
        }

        int d = 0;
        depth[root] = 0;
        bejart[root] = 1;

        List<Integer> BFS = new ArrayList<Integer>();
        BFS.add(root);
        while(!BFS.isEmpty()){
            int e = BFS.remove(0);
            for(int id=0; id<M; id++){
                if(adj(e,id) && bejart[id] ==0){
                    bejart[id] = 1;
                    depth[id] = depth[e] + 1;
                    parent[id] = e;
                    if(depth[id]>d){
                        d++;
                    }
                    BFS.add(id);
                }
            }
        }        
        return d;
    }    
}

class Test{
    public boolean RootedIsomorh(Tree t1, int r1, Tree t2, int r2){

        int M = t1.M;

        int[] label = new int[2*M];
        List<Integer>[] orderlabels = new ArrayList[2*M];
        String[] labelstring = new String[2*M];

        int H1 = t1.depth(r1);
        int H2 = t2.depth(r2);

        if(H1 != H2){
            return false;
        }
        int H = H1;
        
        List<Integer>[] levels;
        levels = new ArrayList[H+1];
        for(int h=0;h<=H;h++){
            levels[h] = new ArrayList<Integer>();
        }

        for(int id=0;id<M;id++){
            levels[t1.depth[id]].add(id);
            levels[t2.depth[id]].add(id+M); // közös listák
        }        

        for(int i=0; i<2*M;i++){ //kezdetben mindenkinek 0 a címkéje
            label[i] = 0;
            orderlabels[i] = new ArrayList();
            labelstring[i] = "0";
        }

        for(int h=H-1;h>=0;h--){ // szintenként

            //Szülők felcímkézése
            for(int j = 0; j < levels[h+1].size();j++){
               int id = levels[h+1].get(j);
               if(id<M){ //t1
                    int p = t1.parent[id];
                    orderlabels[p].add(label[id]);                             
               } else { //t2
                      int p = t2.parent[id-M];
                      orderlabels[p+M].add(label[id]);                      
               }
            }
            // rendezés
            for(int j = 0; j < levels[h].size();j++){
                int k = levels[h].get(j);
                if(orderlabels[k].isEmpty()){
                    labelstring[k]="_";
                } else {
                    labelstring[k]="";
                    Collections.sort(orderlabels[k]);
                    for(Iterator it= orderlabels[k].iterator();it.hasNext();){
                         labelstring[k] +="_"+it.next().toString();
                    }
                }                
            }

            int size = levels[h].size();
            for(int a = 0; a <size-1; a++)
                for(int b = a+1; b<size; b++){
                    if(labelstring[levels[h].get(a)].compareTo(labelstring[levels[h].get(b)])>0){
                       int swap = levels[h].get(a);
                       levels[h].set(a, levels[h].get(b));
                       levels[h].set(b, swap);
                    }
                }

            // címkézés
            int lab = 0;
            label[levels[h].get(0)]= lab;
            String last = labelstring[levels[h].get(0)];
            for(int a = 1; a <size; a++){
                if(!labelstring[levels[h].get(a)].equals(last)){
                    lab++;
                    last = labelstring[levels[h].get(a)];
                }
                label[levels[h].get(a)]= lab;                
            }            
        }        
        return labelstring[r1].equals(labelstring[r2+M]);
    }

    public boolean Isomorh(Tree t1, Tree t2){
        Center C1 = t1.Center();
        Center C2 = t2.Center();
        int R1 = C1.c1;
        
        if(RootedIsomorh(t1,R1,t2,C2.c1)) return true;
        if(C2.db>1){
            if(RootedIsomorh(t1,R1,t2,C2.c2)) return true;
        }
        return false;
    }
}
