package hu.krivan.szakkor.dijkstra;

import java.util.ArrayList;

/**
 * Leír egy lehetséges útvonalat a Gráfon belül
 * 
 * @author Bálint Kriván
 */
public class Trip {

    private int stepCount = 0;
    private int cost = 0;
    /**
     * Eddigi csúcsok (indexei), sorrendben.
     */
    private ArrayList<Integer> nodes;
    /**
     * Még választható csúcsok.
     */
    private ArrayList<Integer> freeNodes;
    private Graph g;

    /**
     * Megadja az első útvonalat, ami igazából csak 1 pontból áll (kezdőpont)
     * @param g Gráf
     */
    public Trip(Graph g) {
        nodes = new ArrayList<Integer>(g.nodes.size());
        freeNodes = new ArrayList<Integer>(g.nodes.size());

        nodes.add(0); // kezdő csúcs
        for (int i = 1; i < g.nodes.size(); i++) {
            freeNodes.add(i);
        }
        this.g = g;
    }

    /**
     * Tulajdonképpen klónozzuk az útvonalat, hogy tudjuk folytatni.
     * @param t Az útvonal amit klónozunk
     */
    private Trip(Trip t) {
        g = t.g;
        stepCount = t.stepCount;
        cost = t.cost;
        // a fenti adatokat könnyű volt másolni, a listákat viszont egyesével kell cloneozni
        // viszont itt nem kell az egyes elemeket cloneozni, mert primitív objektum (int)
        nodes = new ArrayList<Integer>(g.nodes.size());
        freeNodes = new ArrayList<Integer>(g.nodes.size());

        for (int n : t.nodes) {
            nodes.add(n);
        }
        for (int fn : t.freeNodes) {
            freeNodes.add(fn);
        }
    }

    /**
     * Legenerálja az adott útvonal lehetséges folytatásait (csak olyat ad vissza, ami még folytatható)
     * @return
     */
    public ArrayList<Trip> goToNextNodes() {
        ArrayList<Trip> res = new ArrayList<Trip>(freeNodes.size());

        for (int freeNode : freeNodes) {
            Trip t = new Trip(this); // clone
            t.addNode(freeNode); // új pont hozzáadása
            if (!t.outOfSteps()) {
                res.add(t); // csak akkor kell, ha még folytatható, tehát nem léptük át a korlátot.
            }
        }

        return res;
    }

    public boolean isReady() {
        return nodes.get(nodes.size() - 1) == 1; // az utolsó csúcs a végpont.
    }

    public boolean outOfSteps() {
        return stepCount > g.getMaxSteps();
    }

    public boolean cheaperThan(Trip cheapestTrip) {
        if (cheapestTrip == null) {
            return true;
        }

        if (this.getCost() < cheapestTrip.getCost()) {
            return true;
        }

        return false;
    }

    private void addNode(int i) {
        Node newNode = g.nodes.get(i);
        Node prevNode = g.nodes.get(nodes.get(nodes.size() - 1));
        stepCount += newNode.getDistanceFrom(prevNode);
        cost += newNode.getCostFrom(prevNode);

        nodes.add(i);
        freeNodes.remove((Object) i);
    }

    public int getCost() {
        return cost;
    }

    public int getStepCount() {
        return stepCount;
    }

    public void trace() {
        // itt luxus, hogy mindegyik nodeId-ból Node-ot csinálunk
        for( int node : nodes ) {
            if( node != 0 ) {
                System.out.print(" -> ");
            }
            System.out.print(g.nodes.get(node));
        }
        System.out.println();
    }

}
