program saoijd;
uses crt;
const k=10;
      n=11;
var t:array[1..n]of longint;
    i,j:integer;
    a:longint;
begin
     clrscr;
     t[n]:=1;
     for i:=n-1 downto n-k+1 do
     begin
          t[i]:=2*t[i+1];
     end;
     for i:=n-k downto 1 do
     begin
          a:=0;
          for j:=i+1 to i+k do a:=a+t[j];
          t[i]:=a;
     end;
     writeln(t[1]);
     readkey;
end.
