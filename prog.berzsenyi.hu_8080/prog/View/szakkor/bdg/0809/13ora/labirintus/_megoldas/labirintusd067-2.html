﻿/*
﻿ * A http://prog.berzsenyi.hu:8080/prog/View/szakkor/bdg/0809/13ora/labirintus oldalon található Labirintus probléma megoldása.
﻿ * Írta: Kriván Bálint, szakköri megbeszélés alapján (2008. dec. 16-17.)
﻿ */
    
using System;
using System.IO;
using System.Collections.Generic;

namespace Labyrinth
{

    struct Coords
    {
        public int x,y,z;
        
        public Coords( int x1, int y1, int z1 )
        {
            x = x1;
            y = y1;
            z = z1;
        }
    }
    
    /**
     * A labirintus adott mezeje.
     */
    class Mezo
    {
        /* 
            0-üres
            1-fal (ezek között van ajtó is, de egy dimenzióban mindig zárt!)
            2-kapcsoló
         */
        public int v;
        
        /**
         * Lépésszám, alapból -1, azaz még sose jártunk ott
         */
        public int step = -1;
    }
    
    class Labyrinth
    {
        public int n, m;
        public Mezo[,,] table;
        public Queue<Coords> q = new Queue<Coords>();
        public int input;
        
        public Labyrinth(int input)
        {
            this.input = input;
            StreamReader ins = new StreamReader("labi.be"+input);
            string[] line = ins.ReadLine().Split(' ');
            m = int.Parse(line[0]);
            n = int.Parse(line[1]);
            table = new Mezo[n, m, 2];
            byte c;
            for( int i=0; i<m; i++ ) {
                line = ins.ReadLine().Split(' ');
                for( int j=0; j<n; j++ ) {
                    c = byte.Parse( line[j] );
                    table[ j, i, 0 ] = new Mezo();
                    table[ j, i, 1 ] = new Mezo();
                    if( c == 3 ) {
                        table[ j, i, 0 ].v = 0;
                        table[ j, i, 1 ].v = 1;
                    }
                    else if( c == 4 ) {
                        table[ j, i, 0 ].v = 1;
                        table[ j, i, 1 ].v = 0;
                    } else {
                        table[ j, i, 0 ].v = c;
                        table[ j, i, 1 ].v = c;
                    }
                }
            }
            
            solve();
        }
        
        /**
         * Lép az adott (x,y,z)-ről (dx,dy) vektor irányába. Ha lehet, akkor eltároljuk a sorban a lépést, ha nem akkor semmi sem történik.
         */
        public void step( int x, int y, int z, int dx, int dy )
        {
            int prevStep = table[x,y,z].step; //dimenziót mentünk
            if( x+dx >= 0 && x+dx < n && y+dy >= 0 && y+dy < m ) { // ha minden koordináta rendben van
                if( table[ x+dx, y+dy, z ].v != 1 ) { // ha nem falra szereténk lépni
                    if( table[ x+dx, y+dy, z ].v == 2 ) { // ha, ahova lépünk, ott kapcsoló van, akkor dimenzió váltás!
                        z = Math.Abs(1-z);
                    }
                    if( table[ x+dx, y+dy, z ].step == -1 ) { // ha oda lépünk, ahol még nem jártunk
                        // lépésszámot beállítjuk
                        table[ x+dx, y+dy, z ].step = prevStep+1;
                        // sorba betesz
                        q.Enqueue( new Coords( x+dx, y+dy, z ) );
                    }
                }
            }
        }
        
        /**
         * Az adott (x,y,z) pontról ellép (dx,dy) vektorral, ha lehetséges.
         */
        public bool stepBack( ref int x, ref int y, ref int z, int dx, int dy )
        {
            if( x+dx >= 0 && x+dx < n && y+dy >= 0 && y+dy < m ) {
                int prevZ = z;
                if( table[x, y, z].v == 2 ) prevZ = Math.Abs(prevZ-1); // ha egy kapcsoló az őse, akkor annak az előző dimenzióban kell lennie.
                if( table[x+dx, y+dy, z].step == table[x, y, prevZ].step-1 ) {
                    x += dx;
                    y += dy;
                    if( table[x, y, z].v == 2 ) z = Math.Abs(z-1);
                    return true;
                }
            }
            return false;
        }
        
        public void solve()
        {
            // szélességi bejárást tolunk, kezdő a (0,0,0).
            q.Enqueue( new Coords( 0,0,0 ) );
            table[0,0,0].step = 0; // beállítjuk az itten lépésszámot 0-ra.
            Coords c;
            while( q.Count != 0 ) {
                c = q.Dequeue();
                // minden szomszédot checkolunk, ha léphetünk oda, akkor hozzáadjuk a sorhoz.
                step( c.x, c.y, c.z, -1, 0 );
                step( c.x, c.y, c.z, 1, 0 );
                step( c.x, c.y, c.z, 0, -1 );
                step( c.x, c.y, c.z, 0, 1 );
            }
            
            // most kell visszalépegetni, kérdés, hogy melyik dimenzióból indulunk?
            int dim;
            if( table[n-1, m-1, 0].step == -1 )
                if( table[n-1, m-1, 1].step != -1 )
                    dim = 1;
                else
                    dim = -1;
            else if( table[n-1, m-1, 1].step == -1 )
                if( table[n-1, m-1, 0].step != -1 )
                    dim = 0;                
                else
                    dim = -1;
            else if( table[n-1, m-1, 0].step < table[n-1, m-1, 1].step )
                dim = 0;
            else
                dim = 1;
            
            StreamWriter outs = new StreamWriter("labi.ki"+input);
            
            if( dim == -1 ) { // nem találtunk megoldást!
                outs.WriteLine(0);
            } else {
                // van megoldás, és tudjuk, melyik dimenzióban
                // (n-1, m-1, dim)-ből kell visszalépegetni.
                int x = n-1;
                int y = m-1;
                int z = dim;
                int step = table[x, y, z].step;
                outs.WriteLine( step );
                Stack<char> way = new Stack<char>();
                for( int i=0; i<step; i++ ) { // tudjuk, hogy csak step-szer kell lefutnia.
                    if( stepBack( ref x, ref y, ref z, -1, 0) ) {
                        // balra lépünk, de visszafele, azaz valójában jobbra léptünk.
                        way.Push('J');
                        continue;
                    }
                    if( stepBack( ref x, ref y, ref z, +1, 0) ) {
                        way.Push('B');
                        continue;
                    }
                    if( stepBack( ref x, ref y, ref z, 0, -1) ) {
                        way.Push('L');
                        continue;
                    }
                    if( stepBack( ref x, ref y, ref z, 0, +1) ) {
                        way.Push('F');
                        continue;
                    }
                }
                while( way.Count != 0 ) {
                    outs.Write( way.Pop() );
                }
                outs.WriteLine();
            }
            outs.Close();
        }
        
        public static void Main(string[] args)
        {
            //new Labyrinth(0);
            for( int i=1; i<=8; i++ )
                new Labyrinth(i);
        }
    }
}
